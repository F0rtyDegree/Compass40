    87	  void _updateHeading() {
    88	    final now = DateTime.now().millisecondsSinceEpoch;
    89	    _headingSamples.removeWhere((s) => now - s.$2 > _averagingPeriod);
    90	    if (_headingSamples.isEmpty) return;
    91	
    92	    double sinSum = 0, cosSum = 0;
    93	    for (var s in _headingSamples) {
    94	      final a = s.$1 * math.pi / 180;
    95	      sinSum += math.sin(a);
    96	      cosSum += math.cos(a);
    97	    }
    98	    final avgSin = sinSum / _headingSamples.length;
    99	    final avgCos = cosSum / _headingSamples.length;
   100	    final avgHeading = math.atan2(avgSin, avgCos) * 180 / math.pi;
   101	
   102	    double diff = avgHeading - _filteredHeading;
   103	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   104	    _filteredHeading += _smoothingFactor * diff;
   105	    _filteredHeading = (_filteredHeading + 360) % 360;
   106	
   107	    _headingNotifier.value = _filteredHeading;
   108	  }

   247	      return;

   252	    final lastIncompleteEntry = _logItems.lastWhereOrNull(

   256	    if (lastIncompleteEntry != null) {
   257	      final distance = calculateDistance(
   258	        lastIncompleteEntry.latitude,
   259	        lastIncompleteEntry.longitude,
   260	        currentGpsData.latitude!,
   261	        currentGpsData.longitude!,

   279	      id: newId,

   285	      _logItems.add(newEntry);

   287	    });

   290	  }

    52	  void paint(Canvas canvas, Size size) {

    54	    final radius = size.width / 2 - stroke / 2;
    55	    final center = Offset(size.width / 2, size.height / 2);
    56	    final colors = [
    57	      Colors.red,
    58	      Colors.green,
    59	      Colors.yellow,
    60	      Colors.cyan,
    61	      Colors.red,
    62	      Colors.green,
    63	      Colors.yellow,
    64	      Colors.cyan
    65	    ];
    66	    final paint = Paint()
    67	      ..style = PaintingStyle.stroke
    68	      ..strokeWidth = stroke;
    69	
    70	    const sect = math.pi / 4;
    71	    for (int i = 0; i < 8; i++) {
    72	      paint.color = colors[i];
    73	      final start = -math.pi / 2 - sect / 2 + i * sect;
    74	      canvas.drawArc(
    75	          Rect.fromCircle(center: center, radius: radius), start, sect, false, paint);

    77	
    78	    // Разделительные линии (15 градусов)

    80	      ..style = PaintingStyle.stroke
    81	      ..strokeWidth = 1.0;
    82	
    83	    const subSect = math.pi / 12; // 15 градусов
    84	    for (int i = 0; i < 8; i++) {
    85	      final sectorColor = colors[i];
    86	      final lineLuminance = sectorColor.computeLuminance();
    87	      linePaint.color = lineLuminance > 0.5
    88	          ? Colors.black.withAlpha(128)
    89	          : Colors.white.withAlpha(128);
    90	
    91	      final majorStart = -math.pi / 2 - sect / 2 + i * sect;
    92	      final line1Angle = majorStart + subSect;
    93	      final line2Angle = majorStart + 2 * subSect;
    94	
    95	      void drawLineAtAngle(double angle) {
    96	        final startPoint = Offset(
    97	          center.dx + (radius - stroke / 2) * math.cos(angle),
    98	          center.dy + (radius - stroke / 2) * math.sin(angle),
    99	        );
   100	        final endPoint = Offset(
   101	          center.dx + (radius + stroke / 2) * math.cos(angle),
   102	          center.dy + (radius + stroke / 2) * math.sin(angle),
   103	        );
   104	        canvas.drawLine(startPoint, endPoint, linePaint);
   105	      }
   106	
   107	      drawLineAtAngle(line1Angle);
   108	      drawLineAtAngle(line2Angle);
   109	    }
   110	
   111	    // координатные метки N,E,S,W
   112	    final letters = {0: 'С', 2: 'В', 4: 'Ю', 6: 'З'};
   113	    for (var e in letters.entries) {
   114	      final i = e.key;
   115	      final t = e.value;

   117	      final textColor = col.computeLuminance() > 0.5 ? Colors.black : Colors.white;
   118	      final tp = TextPainter(

   129	      canvas.translate(radius, 0);

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) + math.cos(phi1) * math.sin(dR) * math.cos(theta);
    19	  final phi2 = math.asin(sinPhi2);
    20	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    21	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    22	  final lambda2 = lambda1 + math.atan2(y, x);
    23	
    24	  return {
    25	    'latitude': phi2 * 180 / math.pi,
    26	    'longitude': lambda2 * 180 / math.pi,
    27	  };
    28	}
    29	
    30	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    31	  const R = 6371e3;
    32	  final phi1 = lat1 * math.pi / 180;
    33	  final phi2 = lat2 * math.pi / 180;
    34	  final dPhi = (lat2 - lat1) * math.pi / 180;
    35	  final dLambda = (lon2 - lon1) * math.pi / 180;
    36	
    37	  final a = math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    38	      math.cos(phi1) * math.cos(phi2) *
    39	          math.sin(dLambda / 2) * math.sin(dLambda / 2);
    40	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    41	  return R * c;
    42	}
    43	
    44	/// Возвращает истинный азимут
    45	double calculateTrueBearing(double lat1, double lon1, double lat2, double lon2) {
    46	  final phi1 = lat1 * math.pi / 180;
    47	  final phi2 = lat2 * math.pi / 180;
    48	  final dLambda = (lon2 - lon1) * math.pi / 180;
    49	
    50	  final y = math.sin(dLambda) * math.cos(phi2);
    51	  final x =
    52	      math.cos(phi1) * math.sin(phi2) - math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    53	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    54	}

    11	class MyCompassPlugin : FlutterPlugin, EventChannel.StreamHandler, SensorEventListener {
    12	    private lateinit var sensorManager: SensorManager
    13	    private var eventSink: EventChannel.EventSink? = null
    14	    private val accelerometerReading = FloatArray(3)
    15	    private val magnetometerReading = FloatArray(3)
    16	    private val rotationMatrix = FloatArray(9)
    17	    private val orientationAngles = FloatArray(3)
    18	    private var accuracy: Int = SensorManager.SENSOR_STATUS_UNRELIABLE
    19	
    20	    override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
    21	        val context = binding.applicationContext
    22	        sensorManager = context.getSystemService(Context.SENSOR_SERVICE) as SensorManager
    23	        val eventChannel = EventChannel(binding.binaryMessenger, "my_compass/events")
    24	        eventChannel.setStreamHandler(this)
    25	    }
    26	
    27	    override fun onDetachedFromEngine(binding: FlutterPlugin.FlutterPluginBinding) {
    28	        stopListening()
    29	    }
    30	
    31	    override fun onListen(arguments: Any?, events: EventChannel.EventSink?) {
    32	        eventSink = events
    33	        startListening()
    34	    }
    35	
    36	    override fun onCancel(arguments: Any?) {
    37	        stopListening()
    38	        eventSink = null
    39	    }
    40	
    41	    private fun startListening() {
    42	        sensorManager.getDefaultSensor(Sensor.TYPE_ACCELEROMETER)?.also { accelerometer ->
    43	            sensorManager.registerListener(
    44	                this,
    45	                accelerometer,
    46	                SensorManager.SENSOR_DELAY_GAME
    47	            )
    48	        }
    49	        sensorManager.getDefaultSensor(Sensor.TYPE_MAGNETIC_FIELD)?.also { magneticField ->
    50	            sensorManager.registerListener(
    51	                this,
    52	                magneticField,
    53	                SensorManager.SENSOR_DELAY_GAME
    54	            )
    55	        }
    56	    }
    57	
    58	    private fun stopListening() {
    59	        sensorManager.unregisterListener(this)
    60	    }
    61	
    62	    override fun onAccuracyChanged(sensor: Sensor?, accuracy: Int) {
    63	        if (sensor?.type == Sensor.TYPE_MAGNETIC_FIELD) {
    64	            this.accuracy = accuracy
    65	            updateOrientationAngles()
    66	        }
    67	    }
    68	
    69	    override fun onSensorChanged(event: SensorEvent) {
    70	        when (event.sensor.type) {
    71	            Sensor.TYPE_ACCELEROMETER -> {
    72	                System.arraycopy(event.values, 0, accelerometerReading, 0, accelerometerReading.size)
    73	            }
    74	            Sensor.TYPE_MAGNETIC_FIELD -> {
    75	                System.arraycopy(event.values, 0, magnetometerReading, 0, magnetometerReading.size)
    76	            }
    77	        }
    78	        updateOrientationAngles()
    79	    }
    80	
    81	    private fun updateOrientationAngles() {
    82	        SensorManager.getRotationMatrix(rotationMatrix, null, accelerometerReading, magnetometerReading)
    83	        SensorManager.getOrientation(rotationMatrix, orientationAngles)
    84	
    85	        val azimuth = Math.toDegrees(orientationAngles[0].toDouble())
    86	        val heading = (azimuth + 360) % 360

    89	    }
