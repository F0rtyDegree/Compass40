   169	  void _updateHeading() {
   170	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   171	    final gpsBearing = state.gpsBearingNotifier.value;
   172	
   173	    bool useGps = false;
   174	
   175	    switch (state.compassMode) {
   176	      case CompassMode.magnetic:
   177	        useGps = false;
   178	        break;
   179	      case CompassMode.gps:
   180	        useGps = gpsBearing != null;
   181	        break;
   182	      case CompassMode.auto:
   183	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   184	        break;
   185	    }
   186	
   187	    state.isGpsCompassActiveNotifier.value = useGps;
   188	
   189	    if (useGps && gpsBearing != null) {
   190	      // --- ЭТАП 1: Медианная фильтрация ---
   191	      state.gpsBearingSamples.add(gpsBearing);
   192	      if (state.gpsBearingSamples.length > HomeState.maxGpsSamples) {
   193	        state.gpsBearingSamples.removeAt(0);
   194	      }
   195	
   196	      final sortedSamples = List<double>.from(state.gpsBearingSamples)..sort();
   197	      final medianBearing = sortedSamples[sortedSamples.length >> 1]; // Быстрое деление на 2
   198	
   199	      // --- ЭТАП 2: Экспоненциальное сглаживание ---
   200	      if (state.filteredGpsBearing == 0) {
   201	        state.filteredGpsBearing = medianBearing; // Инициализация
   202	      }
   203	      double diff = medianBearing - state.filteredGpsBearing;
   204	      if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   205	
   206	      state.filteredGpsBearing += state.smoothingFactor * diff;
   207	      state.filteredGpsBearing = (state.filteredGpsBearing + 360) % 360;
   208	
   209	      state.headingNotifier.value = state.filteredGpsBearing;
   210	      return;
   211	    } else {
   212	      // Если GPS выключился, сбрасываем фильтр
   213	      state.gpsBearingSamples.clear();
   214	      state.filteredGpsBearing = 0;
   215	    }
   216	
   217	    // Магнитный компас: усреднение по выборке
   218	    final now = DateTime.now().millisecondsSinceEpoch;
   219	    state.headingSamples.removeWhere((s) => now - s.$2 > state.averagingPeriod);
   220	
   221	    if (state.headingSamples.isEmpty) return;
   222	
   223	    double sinSum = 0, cosSum = 0;
   224	    for (var s in state.headingSamples) {
   225	      final a = s.$1 * math.pi / 180;
   226	      sinSum += math.sin(a);
   227	      cosSum += math.cos(a);
   228	    }
   229	
   230	    final avgSin = sinSum / state.headingSamples.length;
   231	    final avgCos = cosSum / state.headingSamples.length;
   232	    final avgHeading = math.atan2(avgSin, avgCos) * 180 / math.pi;
   233	
   234	    double diff = avgHeading - state.filteredHeading;
   235	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   236	
   237	    state.filteredHeading += state.smoothingFactor * diff;
   238	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   239	
   240	    state.headingNotifier.value = state.filteredHeading;
   241	  }

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}

    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}

    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    98	    override fun onAccuracyChanged(sensor: Sensor?, accuracy: Int) {
    99	        when (sensor?.type) {
   100	            Sensor.TYPE_MAGNETIC_FIELD,
   101	            Sensor.TYPE_ROTATION_VECTOR -> {
   102	                this.accuracy = accuracy
   103	            }

   107	    override fun onSensorChanged(event: SensorEvent) {
   108	        when (event.sensor.type) {
   109	            Sensor.TYPE_ROTATION_VECTOR -> {
   110	                if (useRotationVector) {
   111	                    SensorManager.getRotationMatrixFromVector(rotationMatrix, event.values)
   112	                    updateOrientationAnglesFromMatrix(rotationMatrix)
   113	                }
   114	            }
   115	
   116	            Sensor.TYPE_ACCELEROMETER -> {
   117	                if (!useRotationVector) {
   118	                    System.arraycopy(
   119	                        event.values,
   120	                        0,
   121	                        accelerometerReading,
   122	                        0,
   123	                        accelerometerReading.size
   124	                    )
   125	                    updateOrientationAnglesFallback()
   126	                }
   127	            }
   128	
   129	            Sensor.TYPE_MAGNETIC_FIELD -> {
   130	                if (!useRotationVector) {
   131	                    System.arraycopy(
   132	                        event.values,
   133	                        0,
   134	                        magnetometerReading,
   135	                        0,
   136	                        magnetometerReading.size
   137	                    )
   138	                    updateOrientationAnglesFallback()
   139	                }
   140	            }
   141	        }
   142	    }

   144	    private fun updateOrientationAnglesFallback() {
   145	        val success = SensorManager.getRotationMatrix(
   146	            rotationMatrix,
   147	            null,
   148	            accelerometerReading,
   149	            magnetometerReading
   150	        )
   151	
   152	        if (success) {
   153	            updateOrientationAnglesFromMatrix(rotationMatrix)
   154	        }
   155	    }

   157	    private fun updateOrientationAnglesFromMatrix(sourceMatrix: FloatArray) {
   158	        val rotation = getDisplayRotation()
   159	
   160	        val (axisX, axisY) = when (rotation) {
   161	            Surface.ROTATION_90 -> Pair(SensorManager.AXIS_Y, SensorManager.AXIS_MINUS_X)
   162	            Surface.ROTATION_180 -> Pair(SensorManager.AXIS_MINUS_X, SensorManager.AXIS_MINUS_Y)
   163	            Surface.ROTATION_270 -> Pair(SensorManager.AXIS_MINUS_Y, SensorManager.AXIS_X)
   164	            else -> Pair(SensorManager.AXIS_X, SensorManager.AXIS_Y)
   165	        }
   166	
   167	        SensorManager.remapCoordinateSystem(
   168	            sourceMatrix,
   169	            axisX,
   170	            axisY,
   171	            remappedRotationMatrix
   172	        )
   173	
   174	        SensorManager.getOrientation(remappedRotationMatrix, orientationAngles)
   175	
   176	        val azimuthRad = orientationAngles[0].toDouble()
   177	        val azimuthDeg = Math.toDegrees(azimuthRad)
   178	
   179	        val heading = ((azimuthDeg + 360.0) % 360.0)
   180	
   181	        eventSink?.success(listOf(heading, accuracy.toDouble()))
   182	    }
