   172	  double _calculateCircularMedian(List<double> samples) {
   173	    if (samples.isEmpty) return 0.0;
   174	    if (samples.length == 1) return samples[0];
   175	
   176	    samples.sort();
   177	
   178	    double maxGap = 0;
   179	    int maxGapIndex = -1;
   180	
   181	    for (int i = 0; i < samples.length - 1; i++) {
   182	      final gap = samples[i + 1] - samples[i];
   183	      if (gap > maxGap) {
   184	        maxGap = gap;
   185	        maxGapIndex = i;
   186	      }
   187	    }
   188	
   189	    final wrapAroundGap = (samples.first + 360) - samples.last;
   190	    if (wrapAroundGap > maxGap) {
   191	      maxGap = wrapAroundGap;
   192	      maxGapIndex = samples.length - 1;
   193	    }
   194	
   195	    List<double> shiftedSamples;
   196	    if (maxGapIndex == samples.length - 1) {
   197	      shiftedSamples = List.from(samples);
   198	    } else {
   199	      shiftedSamples = [];
   200	      final shiftPoint = samples[maxGapIndex];
   201	      for (final s in samples) {
   202	        if (s > shiftPoint) {
   203	          shiftedSamples.add(s);
   204	        } else {
   205	          shiftedSamples.add(s + 360);
   206	        }
   207	      }
   208	    }
   209	
   210	    double median;
   211	    int mid = shiftedSamples.length ~/ 2;
   212	    if (shiftedSamples.length % 2 == 1) {
   213	      median = shiftedSamples[mid];
   214	    } else {
   215	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   216	    }
   217	
   218	    return median % 360;
   219	  }

   221	  void _updateHeading() {
   222	    final now = DateTime.now().millisecondsSinceEpoch;
   223	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   224	    final gpsBearing = state.gpsBearingNotifier.value;
   225	
   226	    bool useGps = false;
   227	
   228	    switch (state.compassMode) {
   229	      case CompassMode.magnetic:
   230	        useGps = false;
   231	        break;
   232	      case CompassMode.gps:
   233	        useGps = gpsBearing != null;
   234	        break;
   235	      case CompassMode.auto:
   236	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   237	        break;
   238	    }
   239	
   240	    state.isGpsCompassActiveNotifier.value = useGps;
   241	
   242	    double newHeading;
   243	
   244	    if (useGps) {
   245	      // GPS-компас
   246	      state.gpsBearingSamples.removeWhere(
   247	        (s) => now - s.$2 > state.averagingPeriod,
   248	      );
   249	      if (state.gpsBearingSamples.isEmpty) return;
   250	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   251	      final medianTrueBearing = _calculateCircularMedian(bearings);
   252	      // Преобразуем истинный курс в магнитный
   253	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   254	    } else {
   255	      // Магнитный компас
   256	      state.headingSamples.removeWhere(
   257	        (s) => now - s.$2 > state.averagingPeriod,
   258	      );
   259	      if (state.headingSamples.isEmpty) return;
   260	      final headings = state.headingSamples.map((s) => s.$1).toList();
   261	      // Данные с сенсора уже магнитные
   262	      newHeading = _calculateCircularMedian(headings);
   263	    }
   264	
   265	    // Сглаживание
   266	    double diff = newHeading - state.filteredHeading;
   267	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   268	
   269	    state.filteredHeading += state.smoothingFactor * diff;
   270	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   271	
   272	    state.headingNotifier.value = state.filteredHeading;
   273	  }

   279	  void _calculateWaypointData() {

   332	    final sin = math.sin(angle);
   333	    final derotated = Offset(
   334	      relative.dx * cos - relative.dy * sin,
   335	      relative.dx * sin + relative.dy * cos,
   336	    );
   337	
   338	    final unscaled = derotated / t.scale;
   339	    return unscaled + Offset(imageSize.width / 2, imageSize.height / 2);
   340	  }
   341	
   342	  Offset imageToScreen(Offset imagePoint) {
   343	    if (state.imageSize == null || state.viewportSize == null) {
   344	      return imagePoint;
   345	    }
   346	
   347	    final vp = state.viewportSize!;
   348	    final t = state.transformState;
   349	    final imageSize = state.imageSize!;
   350	
   351	    final center = Offset(vp.width / 2, vp.height / 2);
   352	    final local =
   353	        imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   354	    final scaled = local * t.scale;
   355	
   356	    final angle = t.rotationRadians;
   357	    final cos = math.cos(angle);
   358	    final sin = math.sin(angle);
   359	    final rotated = Offset(
   360	      scaled.dx * cos - scaled.dy * sin,
   361	      scaled.dx * sin + scaled.dy * cos,
   362	    );
   363	
   364	    return center + t.translation + rotated;
   365	  }

   428	  Future<void> _addAnchor({
   429	    required Offset imagePoint,
   430	    required double latitude,
   431	    required double longitude,
   432	  }) async {
   433	    final project = state.project;
   434	    if (project == null) return;
   435	
   436	    final newPathJumpIndices = [...project.pathJumpIndices];
   437	    if (project.userPath.isNotEmpty) {
   438	      newPathJumpIndices.add(project.userPath.length);
   439	    }
   440	
   441	    double? distanceFromPrevious;
   442	    if (project.anchors.isNotEmpty) {
   443	      final lastAnchor = project.anchors.last;
   444	      distanceFromPrevious = calibration.distanceBetweenAnchorsMeters(lastAnchor, MapAnchor(
   445	        id: '',
   446	        imageX: imagePoint.dx,
   447	        imageY: imagePoint.dy,
   448	        latitude: latitude,
   449	        longitude: longitude,
   450	        createdAt: DateTime.now(),
   451	      ));
   452	    }
   453	
   454	    final anchor = MapAnchor(
   455	      id: DateTime.now().millisecondsSinceEpoch.toString(),
   456	      imageX: imagePoint.dx,
   457	      imageY: imagePoint.dy,
   458	      latitude: latitude,
   459	      longitude: longitude,
   460	      createdAt: DateTime.now(),
   461	    );
   462	
   463	    final updatedAnchors = [...project.anchors, anchor];
   464	    final updatedProject = project.copyWith(
   465	      anchors: updatedAnchors,
   466	      pathJumpIndices: newPathJumpIndices,
   467	    );

   469	    await storageService.saveProject(updatedProject);
   470

   475	    _recalculateWorkingPairAndRotation();
   476	    _recalculateCanPlaceTarget();
   477	    _recalculateUserImagePoint();

   488	  }

   864	  void _applyHeadingRotation() {
   865	    if (!state.followMode || state.heading == null) return;
   866	
   867	    _isAutoRotating = true;
   868	
   869	    // True North = Magnetic North + Declination
   870	    final magneticHeadingRad = (state.heading!) * (math.pi / 180);
   871	    final declinationRad = magneticDeclination * (math.pi / 180);
   872	    final trueHeadingRad = magneticHeadingRad + declinationRad;
   873	
   874	    final targetRotation = -trueHeadingRad + state.mapRotation;
   875	
   876	    final current = state.transformState;
   877	    final pivotImage = screenToImage(_getCrosshairScreenPoint());
   878	
   879	    final tempTransform = current.copyWith(rotationRadians: targetRotation);
   880	    final oldTransform = state.transformState;
   881	    state.transformState = tempTransform;
   882	    final pivotScreenAfterRotate = imageToScreen(pivotImage);
   883	    state.transformState = oldTransform;
   884	
   885	    final delta = _getCrosshairScreenPoint() - pivotScreenAfterRotate;
   886	    final newTranslation = current.translation + delta;
   887	
   888	    updateTransform(
   889	      current.copyWith(
   890	        rotationRadians: targetRotation,
   891	        translation: newTranslation,
   892	      ),
   893	    );
   894	
   895	    _isAutoRotating = false;
   896	  }

   302	  void _scaleAroundCrosshair(MapTransformState current, double newScale) {

   305	    final pivotScreen = _logic.getCrosshairScreenPoint();

   309	      scale: newScale,

   311	      translation: current.translation,
   312	    );
   313	
   314	    final oldTransform = _state.transformState;

   317	    final pivotScreenAfterScale = _logic.imageToScreen(pivotImage);
   318	
   319	    _state.transformState = oldTransform;
   320	
   321	    final delta = pivotScreen - pivotScreenAfterScale;
   322	    final newTranslation = current.translation + delta;
   323	
   324	    _logic.updateTransform(
   325	      MapTransformState(
   326	        scale: newScale,

    29	  MapWorkingPair? selectWorkingPair(
    30	    List<MapAnchor> anchors, {
    31	    double minDistanceMeters = 50.0,
    32	  }) {
    33	    if (anchors.length < 2) return null;
    34	
    35	    final sorted = List<MapAnchor>.from(anchors)
    36	      ..sort((a, b) => a.createdAt.compareTo(b.createdAt));
    37	
    38	    final latest = sorted.last;
    39	
    40	    for (int i = sorted.length - 2; i >= 0; i--) {
    41	      final candidate = sorted[i];
    42	      final dist = distanceBetweenAnchorsMeters(latest, candidate);
    43	      if (dist >= minDistanceMeters) {
    44	        return MapWorkingPair(latest: latest, reference: candidate);
    45	      }
    46	    }
    47	
    48	    return null;
    49	  }

    57	  // Расстояние между точками привязки
    58	  // ---------------------------------------------------------
    59	
    60	  double distanceBetweenAnchorsMeters(MapAnchor a, MapAnchor b) {
    61	    return _haversineDistance(a.latitude, a.longitude, b.latitude, b.longitude);
    62	  }
    63	
    64	  double _haversineDistance(
    65	    double lat1,
    66	    double lon1,
    67	    double lat2,
    68	    double lon2,
    69	  ) {
    70	    final phi1 = lat1 * math.pi / 180;
    71	    final phi2 = lat2 * math.pi / 180;
    72	    final dPhi = (lat2 - lat1) * math.pi / 180;
    73	    final dLambda = (lon2 - lon1) * math.pi / 180;
    74	
    75	    final a =
    76	        math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    77	        math.cos(phi1) *
    78	            math.cos(phi2) *
    79	            math.sin(dLambda / 2) *
    80	            math.sin(dLambda / 2);
    81	    final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    82	
    83	    return _earthRadius * c;
    84	  }
    85	
    86	  // ---------------------------------------------------------
    87	  // Перевод GPS → LocalPoint (метры East/North)
    88	  // ---------------------------------------------------------
    89	
    90	  LocalPoint geoToLocal({
    91	    required double latitude,
    92	    required double longitude,
    93	    required double originLat,
    94	    required double originLon,
    95	  }) {
    96	    final dLat = (latitude - originLat) * math.pi / 180;
    97	    final dLon = (longitude - originLon) * math.pi / 180;
    98	    final originLatRad = originLat * math.pi / 180;
    99	
   100	    final north = dLat * _earthRadius;
   101	    final east = dLon * _earthRadius * math.cos(originLatRad);
   102	
   103	    return LocalPoint(east: east, north: north);
   104	  }
   105	
   106	  GeoPoint localToGeo({
   107	    required LocalPoint local,
   108	    required double originLat,
   109	    required double originLon,
   110	  }) {
   111	    final originLatRad = originLat * math.pi / 180;
   112	
   113	    final dLat = local.north / _earthRadius;
   114	    final dLon = local.east / (_earthRadius * math.cos(originLatRad));
   115	
   116	    return GeoPoint(
   117	      latitude: originLat + dLat * 180 / math.pi,
   118	      longitude: originLon + dLon * 180 / math.pi,
   119	    );
   120	  }

   122	  // ---------------------------------------------------------
   123	  // Построение локального 2-точечного преобразования
   124	  // ---------------------------------------------------------
   125	
   126	  _SimilarityTransform? _buildTransform(MapWorkingPair pair) {
   127	    final imgVec = Offset(
   128	      pair.latest.imageX - pair.reference.imageX,
   129	      pair.latest.imageY - pair.reference.imageY,
   130	    );
   131	
   132	    final imgLen = imgVec.distance;
   133	    if (imgLen < 1e-9) return null;
   134	
   135	    final originLat = pair.reference.latitude;
   136	    final originLon = pair.reference.longitude;
   137	
   138	    final latestLocal = geoToLocal(
   139	      latitude: pair.latest.latitude,
   140	      longitude: pair.latest.longitude,
   141	      originLat: originLat,
   142	      originLon: originLon,
   143	    );
   144	
   145	    final geoVec = Offset(latestLocal.east, -latestLocal.north);
   146	    final geoLen = geoVec.distance;
   147	    if (geoLen < 1e-9) return null;
   148	
   149	    final scale = geoLen / imgLen;
   150	
   151	    final angle =
   152	        math.atan2(geoVec.dy, geoVec.dx) - math.atan2(imgVec.dy, imgVec.dx);
   153	
   154	    return _SimilarityTransform(
   155	      originLat: originLat,
   156	      originLon: originLon,
   157	      referenceImageX: pair.reference.imageX,
   158	      referenceImageY: pair.reference.imageY,
   159	      scale: scale,
   160	      angleRadians: angle,
   161	    );
   162	  }
   163	
   164	  // ---------------------------------------------------------
   165	  // image ↔ GPS
   166	  // ---------------------------------------------------------
   167	
   168	  GeoPoint? imagePointToGeo({
   169	    required Offset imagePoint,
   170	    required MapWorkingPair pair,
   171	  }) {
   172	    final t = _buildTransform(pair);
   173	    if (t == null) return null;
   174	
   175	    final dx = imagePoint.dx - t.referenceImageX;
   176	    final dy = imagePoint.dy - t.referenceImageY;
   177	
   178	    final scaledDx = dx * t.scale;
   179	    final scaledDy = dy * t.scale;
   180	
   181	    final cos = math.cos(t.angleRadians);
   182	    final sin = math.sin(t.angleRadians);
   183	    final localEast = scaledDx * cos - scaledDy * sin;
   184	    final localNorthNeg = scaledDx * sin + scaledDy * cos;
   185	
   186	    final localNorth = -localNorthNeg;
   187	
   188	    return localToGeo(
   189	      local: LocalPoint(east: localEast, north: localNorth),
   190	      originLat: t.originLat,
   191	      originLon: t.originLon,
   192	    );
   193	  }
   194	
   195	  Offset? geoToImagePoint({
   196	    required double latitude,
   197	    required double longitude,
   198	    required MapWorkingPair pair,
   199	  }) {
   200	    final t = _buildTransform(pair);
   201	    if (t == null) return null;
   202	
   203	    final local = geoToLocal(
   204	      latitude: latitude,
   205	      longitude: longitude,
   206	      originLat: t.originLat,
   207	      originLon: t.originLon,
   208	    );
   209	
   210	    final geoX = local.east;
   211	    final geoY = -local.north;
   212	
   213	    final cos = math.cos(-t.angleRadians);
   214	    final sin = math.sin(-t.angleRadians);
   215	    final rotX = geoX * cos - geoY * sin;
   216	    final rotY = geoX * sin + geoY * cos;
   217	
   218	    final dx = rotX / t.scale;
   219	    final dy = rotY / t.scale;
   220	
   221	    return Offset(t.referenceImageX + dx, t.referenceImageY + dy);
   222	  }

   225	  // Вспомогательные проверки
   226	  // ---------------------------------------------------------

   228	  bool canBuildTransform(List<MapAnchor> anchors) {
   229	    return selectWorkingPair(anchors) != null;

   232	  BearingAndDistance bearingAndDistance({
   233	    required double fromLat,
   234	    required double fromLon,
   235	    required double toLat,
   236	    required double toLon,
   237	    required double magneticDeclination,
   238	  }) {
   239	    final dist = _haversineDistance(fromLat, fromLon, toLat, toLon);
   240	
   241	    final phi1 = fromLat * math.pi / 180;
   242	    final phi2 = toLat * math.pi / 180;
   243	    final dLambda = (toLon - fromLon) * math.pi / 180;
   244	
   245	    final y = math.sin(dLambda) * math.cos(phi2);
   246	    final x = 
   247	        math.cos(phi1) * math.sin(phi2) -
   248	        math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
   249	
   250	    final trueBearing = (math.atan2(y, x) * 180 / math.pi + 360) % 360;
   251	    // Convention: East declination is positive. True North = Magnetic North + Declination.
   252	    final magneticBearing = (trueBearing - magneticDeclination + 360) % 360;
   253	
   254	    return BearingAndDistance(
   255	      distanceMeters: dist,
   256	      trueBearing: trueBearing,
   257	      magneticBearing: magneticBearing,
   258	    );
   259	  }
   260	}

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}
    30	
    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}
    47	
    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    72	    _gestureStartTranslation = widget.transformState.translation;
    73	    _gestureStartScale = widget.transformState.scale;
    74	    _gestureStartRotation = widget.transformState.rotationRadians;
    75	    _gestureStartFocalPoint = details.focalPoint;

    79	  void _onScaleUpdate(ScaleUpdateDetails details, Size viewportSize) {
    80	    final center = Offset(viewportSize.width / 2, viewportSize.height / 2);
    81	
    82	    // Смещение фокальной точки с момента начала жеста
    83	    final focalDelta = details.focalPoint - _gestureStartFocalPoint;
    84	
    85	    // Новый масштаб
    86	    final newScale = (_gestureStartScale * details.scale).clamp(0.05, 20.0);
    87	
    88	    // Поворот: изменение угла с начала жеста
    89	    final deltaRotation = details.rotation - _previousGestureRotation;
    90	    _previousGestureRotation = details.rotation;
    91	    final newRotation = widget.transformState.rotationRadians + deltaRotation;
    92	
    93	    // Смещение: комбинация pan + сохранение центра масштаба/поворота
    94	    // Пересчитываем translation так, чтобы focal point оставался на месте
    95	    final scaleRatio = newScale / _gestureStartScale;
    96	
    97	    // Вектор от центра viewport до фокальной точки в начале жеста
    98	    final focalFromCenter = _gestureStartFocalPoint - center;
    99	
   100	    // После масштаба + поворота этот вектор увеличивается
   101	    final rotDelta = newRotation - _gestureStartRotation;
   102	    final cos = math.cos(rotDelta);
   103	    final sin = math.sin(rotDelta);
   104	
   105	    final rotatedFocal = Offset(
   106	      focalFromCenter.dx * cos - focalFromCenter.dy * sin,
   107	      focalFromCenter.dx * sin + focalFromCenter.dy * cos,
   108	    );
   109	
   110	    final newTranslation =
   111	        _gestureStartTranslation +
   112	        focalDelta -
   113	        (rotatedFocal * scaleRatio - focalFromCenter);

   115	    widget.onTransformChanged(
   116	      MapTransformState(
   117	        scale: newScale,
   118	        rotationRadians: newRotation,

   235	  void _drawDashedLine(Canvas canvas, Offset from, Offset to, Paint paint) {
   236	    const dashLength = 10.0;
   237	    const gapLength = 6.0;
   238	    final totalLength = (to - from).distance;
   239	    if (totalLength < 1) return;
   240	
   241	    final direction = (to - from) / totalLength;
   242	    double drawn = 0;
   243	    bool drawing = true;
   244	
   245	    while (drawn < totalLength) {
   246	      final segLen = drawing ? dashLength : gapLength;
   247	      final end = math.min(drawn + segLen, totalLength);
   248	      if (drawing) {
   249	        canvas.drawLine(from + direction * drawn, from + direction * end, paint);
   250	      }
   251	      drawn = end;
   252	      drawing = !drawing;
   253	    }
   254	  }
