   157	  double _calculateCircularMedian(List<double> samples) {
   158	    if (samples.isEmpty) return 0.0;
   159	    if (samples.length == 1) return samples[0];
   160	
   161	    samples.sort();
   162	
   163	    double maxGap = 0;
   164	    int maxGapIndex = -1;
   165	
   166	    for (int i = 0; i < samples.length - 1; i++) {
   167	      final gap = samples[i + 1] - samples[i];
   168	      if (gap > maxGap) {
   169	        maxGap = gap;
   170	        maxGapIndex = i;
   171	      }
   172	    }
   173	
   174	    final wrapAroundGap = (samples.first + 360) - samples.last;
   175	    if (wrapAroundGap > maxGap) {
   176	      maxGap = wrapAroundGap;
   177	      maxGapIndex = samples.length - 1;
   178	    }
   179	
   180	    List<double> shiftedSamples;
   181	    if (maxGapIndex == samples.length - 1) {
   182	      shiftedSamples = List.from(samples);
   183	    } else {
   184	      shiftedSamples = [];
   185	      final shiftPoint = samples[maxGapIndex];
   186	      for (final s in samples) {
   187	        if (s > shiftPoint) {
   188	          shiftedSamples.add(s);
   189	        } else {
   190	          shiftedSamples.add(s + 360);
   191	        }
   192	      }
   193	    }
   194	
   195	    double median;
   196	    int mid = shiftedSamples.length ~/ 2;
   197	    if (shiftedSamples.length % 2 == 1) {
   198	      median = shiftedSamples[mid];
   199	    } else {
   200	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   201	    }
   202	
   203	    return median % 360;
   204	  }

   206	  void _updateHeading() {
   207	    final now = DateTime.now().millisecondsSinceEpoch;
   208	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   209	    final gpsBearing = state.gpsBearingNotifier.value;
   210	
   211	    bool useGps = false;
   212	
   213	    switch (state.compassMode) {
   214	      case CompassMode.magnetic:
   215	        useGps = false;
   216	        break;
   217	      case CompassMode.gps:
   218	        useGps = gpsBearing != null;
   219	        break;
   220	      case CompassMode.auto:
   221	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   222	        break;
   223	    }
   224	
   225	    state.isGpsCompassActiveNotifier.value = useGps;
   226	
   227	    double newHeading;
   228	
   229	    if (useGps) {
   230	      state.gpsBearingSamples.removeWhere(
   231	        (s) => now - s.$2 > state.averagingPeriod,
   232	      );
   233	      if (state.gpsBearingSamples.isEmpty) return;
   234	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   235	      final medianTrueBearing = _calculateCircularMedian(bearings);
   236	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   237	    } else {
   238	      state.headingSamples.removeWhere(
   239	        (s) => now - s.$2 > state.averagingPeriod,
   240	      );
   241	      if (state.headingSamples.isEmpty) return;
   242	      final headings = state.headingSamples.map((s) => s.$1).toList();
   243	      newHeading = _calculateCircularMedian(headings);
   244	    }
   245	
   246	    double diff = newHeading - state.filteredHeading;
   247	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   248	
   249	    state.filteredHeading += state.smoothingFactor * diff;
   250	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   251	
   252	    state.headingNotifier.value = state.filteredHeading;
   253	  }

   302	  Offset screenToImage(Offset screenPoint) {
   303	    if (state.imageSize == null || state.viewportSize == null) {
   304	      return screenPoint;
   305	    }
   306	
   307	    final vp = state.viewportSize!;
   308	    final t = state.transformState;
   309	    final imageSize = state.imageSize!;
   310	
   311	    final center = Offset(vp.width / 2, vp.height / 2);
   312	    final relative = screenPoint - center - t.translation;
   313	
   314	    final angle = -t.rotationRadians;
   315	    final cos = math.cos(angle);
   316	    final sin = math.sin(angle);
   317	    final derotated = Offset(
   318	      relative.dx * cos - relative.dy * sin,
   319	      relative.dx * sin + relative.dy * cos,
   320	    );
   321	
   322	    final unscaled = derotated / t.scale;
   323	    return unscaled + Offset(imageSize.width / 2, imageSize.height / 2);
   324	  }

   327	    if (state.imageSize == null || state.viewportSize == null) {
   328	      return imagePoint;
   329	    }

   331	    final vp = state.viewportSize!;
   332	    final t = state.transformState;
   333	    final imageSize = state.imageSize!;
   334	
   335	    final center = Offset(vp.width / 2, vp.height / 2);
   336	    final local = imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   337	    final scaled = local * t.scale;
   338	
   339	    final angle = t.rotationRadians;
   340	    final cos = math.cos(angle);
   341	    final sin = math.sin(angle);
   342	    final rotated = Offset(
   343	      scaled.dx * cos - scaled.dy * sin,
   344	      scaled.dx * sin + scaled.dy * cos,
   345	    );
   346	
   347	    return center + t.translation + rotated;

   762	  void _centerMapOnUser() {
   763	    final imagePoint = state.currentUserImagePoint;
   764	    if (imagePoint == null || state.viewportSize == null) return;
   765	
   766	    final crosshairScreen = _getCrosshairScreenPoint();
   767	    final vp = state.viewportSize!;
   768	    final vpCenter = Offset(vp.width / 2, vp.height / 2);
   769	    final targetTranslation = crosshairScreen - vpCenter;
   770	
   771	    final t = state.transformState;
   772	    final imageSize = state.imageSize;
   773	    if (imageSize == null) return;
   774	
   775	    final local = imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   776	    final scaled = local * t.scale;
   777	
   778	    final angle = t.rotationRadians;
   779	    final cos = math.cos(angle);
   780	    final sin = math.sin(angle);
   781	    final rotated = Offset(
   782	      scaled.dx * cos - scaled.dy * sin,
   783	      scaled.dx * sin + scaled.dy * cos,
   784	    );
   785	
   786	    final newTranslation = targetTranslation - rotated;
   787	
   788	    setState(() {
   789	      state.transformState = t.copyWith(translation: newTranslation);
   790	    });
   791	
   792	    _recalculateUserScreenPoint();
   793	  }

   837	  void _applyHeadingRotation() {

   839	
   840	    _isAutoRotating = true;
   841	
   842	    final magneticHeadingRad = (state.heading!) * (math.pi / 180);
   843	    final declinationRad = magneticDeclination * (math.pi / 180);
   844	    final trueHeadingRad = magneticHeadingRad + declinationRad;
   845

   847	
   848	    final current = state.transformState;
   849	    final pivotImage = screenToImage(_getCrosshairScreenPoint());
   850	
   851	    final tempTransform = current.copyWith(rotationRadians: targetRotation);
   852	    final oldTransform = state.transformState;
   853	    state.transformState = tempTransform;
   854	    final pivotScreenAfterRotate = imageToScreen(pivotImage);
   855	    state.transformState = oldTransform;
   856	
   857	    final delta = _getCrosshairScreenPoint() - pivotScreenAfterRotate;
   858	    final newTranslation = current.translation + delta;
   859	
   860	    updateTransform(
   861	      current.copyWith(
   862	        rotationRadians: targetRotation,
   863	        translation: newTranslation,
   864	      ),
   865	    );
   866	
   867	    _isAutoRotating = false;
   868	  }

   347	    _gestureStartScale = _state.transformState.scale;
   348	    _gestureStartRotation = _state.transformState.rotationRadians;
   349	    _gestureStartFocalPoint = details.focalPoint;

   359	      final startVector = _gestureStartFocalPoint - pivotScreen;

   366	  void _onScaleUpdate(ScaleUpdateDetails details) {
   367	    if (_state.viewportSize == null || _state.imageSize == null) return;
   368	    final pointerCount = details.pointerCount;
   369	    if (pointerCount == 1) {
   370	      if (_state.rotateMode) {
   371	        final currentFocalPoint = details.focalPoint;
   372	        final pivotScreen = _logic.getCrosshairScreenPoint();
   373	        final currentVector = currentFocalPoint - pivotScreen;
   374	        final currentAngle = math.atan2(currentVector.dy, currentVector.dx);
   375	        double deltaAngle = currentAngle - _lastAngle;
   376	        if (deltaAngle > math.pi) deltaAngle -= 2 * math.pi;
   377	        if (deltaAngle < -math.pi) deltaAngle += 2 * math.pi;
   378	        _accumulatedRotation += deltaAngle;
   379	        _lastAngle = currentAngle;
   380	        const sensitivity = 0.8;
   381	        final newRotation = _gestureStartRotation + _accumulatedRotation * sensitivity;
   382	        if (_gestureStartPivotImage != null) {
   383	          final tempTransform = MapTransformState(
   384	            scale: _state.transformState.scale,
   385	            rotationRadians: newRotation,
   386	            translation: _gestureStartTranslation,
   387	          );
   388	          final oldTransform = _state.transformState;
   389	          _state.transformState = tempTransform;
   390	          final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   391	          _state.transformState = oldTransform;
   392	          final delta = pivotScreen - pivotScreenAfterRotate;
   393	          final newTranslation = _gestureStartTranslation + delta;
   394	          _logic.updateTransform(
   395	            MapTransformState(
   396	              scale: _state.transformState.scale,
   397	              rotationRadians: newRotation,
   398	              translation: newTranslation,
   399	            ),
   400	          );
   401	        } else {
   402	          _logic.updateTransform(
   403	            MapTransformState(
   404	              scale: _state.transformState.scale,
   405	              rotationRadians: newRotation,
   406	              translation: _state.transformState.translation,
   407	            ),
   408	          );
   409	        }
   410	      } else {
   411	        final delta = details.focalPoint - _gestureStartFocalPoint;
   412	        final newTranslation = _gestureStartTranslation + delta;
   413	        _logic.updateTransform(
   414	          MapTransformState(
   415	            scale: _state.transformState.scale,
   416	            rotationRadians: _state.transformState.rotationRadians,
   417	            translation: newTranslation,
   418	          ),
   419	        );
   420	      }
   421	      return;
   422	    }
   423	    if (pointerCount == 2) {
   424	      final scaleChange = (details.scale - 1.0).abs();
   425	      final rotationChange = details.rotation.abs();
   426	      if (scaleChange > rotationChange) {

   214	    required double toLat,
   215	    required double toLon,
   216	    required double magneticDeclination,
   217	  }) {
   218	    final dist = calculateDistance(fromLat, fromLon, toLat, toLon);
   219	
   220	    final phi1 = fromLat * math.pi / 180;
   221	    final phi2 = toLat * math.pi / 180;
   222	    final dLambda = (toLon - fromLon) * math.pi / 180;
   223	
   224	    final y = math.sin(dLambda) * math.cos(phi2);
   225	    final x = 
   226	        math.cos(phi1) * math.sin(phi2) -
   227	        math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
   228	
   229	    final trueBearing = (math.atan2(y, x) * 180 / math.pi + 360) % 360;
   230	    final magneticBearing = (trueBearing - magneticDeclination + 360) % 360;
   231	
   232	    return BearingAndDistance(
   233	      distanceMeters: dist,
   234	      trueBearing: trueBearing,
   235	      magneticBearing: magneticBearing,
   236	    );
   237	  }
   238	}

     1	import 'dart:math' as math;

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}
    30	
    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}
    47	
    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    79	  void _onScaleUpdate(ScaleUpdateDetails details, Size viewportSize) {
    80	    final center = Offset(viewportSize.width / 2, viewportSize.height / 2);
    81	
    82	    // Смещение фокальной точки с момента начала жеста
    83	    final focalDelta = details.focalPoint - _gestureStartFocalPoint;
    84	
    85	    // Новый масштаб
    86	    final newScale = (_gestureStartScale * details.scale).clamp(0.05, 20.0);

    90	    _previousGestureRotation = details.rotation;

    92	
    93	    // Смещение: комбинация pan + сохранение центра масштаба/поворота

    95	    final scaleRatio = newScale / _gestureStartScale;

    98	    final focalFromCenter = _gestureStartFocalPoint - center;

   103	    final sin = math.sin(rotDelta);

   111	        _gestureStartTranslation +

   132	  _MapPainter({

    10	  final Size viewportSize;
    11	
    12	  final List<MapAnchor> anchors;
    13	  final List<MapTarget> targets;
    14	  final List<Offset> userPath;
    15	  final List<int> pathJumpIndices;

    17	  final Offset? currentUserImagePoint;
    18	  final Offset? activeTargetImagePoint;

    21	  final double? previewBearingDegrees;
    22	  final double? heading;
    23	  final double mapRotation;
    24	  final double magneticDeclination;

    26	  const MapOverlayPainter({
    27	    required this.imageSize,
    28	    required this.transformState,
    29	    required this.viewportSize,
    30	    required this.anchors,
    31	    required this.targets,
    32	    this.userPath = const [],
    33	    this.pathJumpIndices = const [],
    34	    this.currentUserImagePoint,

    43	  @override
    44	  void paint(Canvas canvas, Size size) {
    45	    _drawUserPath(canvas);
    46	
    47	    for (final anchor in anchors) {
    48	      final screen = imageToScreen(Offset(anchor.imageX, anchor.imageY));
    49	      _drawAnchor(canvas, screen);
    50	    }
    51	
    52	    for (final target in targets) {
    53	      final screen = imageToScreen(Offset(target.imageX, target.imageY));
    54	      _drawTarget(canvas, screen, target.status);
    55	    }
    56	
    57	    if (currentUserImagePoint != null) {
    58	      final screen = imageToScreen(currentUserImagePoint!);
    59	
    60	      if (activeTargetImagePoint != null) {
    61	        final targetScreen = imageToScreen(activeTargetImagePoint!);
    62	        _drawLine(canvas, screen, targetScreen);
    63	        if (previewDistanceMeters != null && previewBearingDegrees != null) {
    64	          _drawLabels(
    65	            canvas,
    66	            screen,
    67	            targetScreen,
    68	            previewDistanceMeters!,
    69	            previewBearingDegrees!,
    70	          );
    71	        }
    72	      }
    73	      _drawCurrentPosition(canvas, screen);
    74	    }
    75	  }
    76	  
    77	  void _drawUserPath(Canvas canvas) {
    78	    if (userPath.length < 2) return;
    79	
    80	    final solidPathPaint = Paint()
    81	      ..color = Colors.blue.withAlpha(204)
    82	      ..strokeWidth = 3.0
    83	      ..style = PaintingStyle.stroke
    84	      ..strokeJoin = StrokeJoin.round
    85	      ..strokeCap = StrokeCap.round;
    86	
    87	    final dashedPathPaint = Paint()
    88	      ..color = Colors.blue.withAlpha(150)
    89	      ..strokeWidth = 2.5
    90	      ..style = PaintingStyle.stroke
    91	      ..strokeCap = StrokeCap.round;
    92	      
    93	    final jumpIndicesSet = pathJumpIndices.toSet();
    94	
    95	    for (int i = 1; i < userPath.length; i++) {
    96	      final p1 = imageToScreen(userPath[i - 1]);
    97	      final p2 = imageToScreen(userPath[i]);
    98	
    99	      if (jumpIndicesSet.contains(i)) {
   100	        _drawDashedLine(canvas, p1, p2, dashedPathPaint);
   101	      } else {
   102	        canvas.drawLine(p1, p2, solidPathPaint);
   103	      }
   104	    }
   105	  }
   106	
   107	  Offset imageToScreen(Offset imagePoint) {
   108	    final center = Offset(viewportSize.width / 2, viewportSize.height / 2);
   109	    final local = imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   110	    final scaled = local * transformState.scale;
   111	    final angle = transformState.rotationRadians;
   112	    final cos = math.cos(angle);
   113	    final sin = math.sin(angle);
   114	    final rotated = Offset(
   115	      scaled.dx * cos - scaled.dy * sin,
   116	      scaled.dx * sin + scaled.dy * cos,
   117	    );
   118	    return center + transformState.translation + rotated;
   119	  }

   121	  void _drawAnchor(Canvas canvas, Offset screen) {
   122	    final paint = Paint()
   123	      ..color = Colors.purple
   124	      ..style = PaintingStyle.fill;

   129	      ..strokeWidth = 2.0;

   133	      screen,
   134	      10,

   136	        ..style = PaintingStyle.stroke
   137	        ..strokeWidth = 2.0,

   141	      ..color = Colors.purple
   142	      ..strokeWidth = 2.0
   143	      ..style = PaintingStyle.stroke;
   144	
   145	    canvas.drawLine(screen + const Offset(-5, 0), screen + const Offset(5, 0), linePaint);
   146	    canvas.drawLine(screen + const Offset(0, -8), screen + const Offset(0, 5), linePaint);
   147	
   148	    final path = Path()
   149	      ..moveTo(screen.dx - 5, screen.dy + 5)
   150	      ..lineTo(screen.dx, screen.dy + 10)
   151	      ..lineTo(screen.dx + 5, screen.dy + 5);
   152	    canvas.drawPath(path, linePaint);
   153	  }

   155	  void _drawCurrentPosition(Canvas canvas, Offset screen) {
   156	    final outlinePaint = Paint()
   157	      ..color = Colors.blue.shade500
   158	      ..strokeWidth = 2.0
   159	      ..style = PaintingStyle.stroke
   160	      ..strokeJoin = StrokeJoin.round;
   161	
   162	    final innerShadowPaint = Paint()
   163	      ..color = Colors.black.withAlpha(128)
   164	      ..strokeWidth = 0.75
   165	      ..style = PaintingStyle.stroke
   166	      ..strokeJoin = StrokeJoin.round;

   169	      ..moveTo(0, -30)
   170	      ..lineTo(15, 15)
   171	      ..lineTo(0, 10)
   172	      ..lineTo(-15, 15)
   173	      ..close();
   174	
   175	    canvas.save();
   176	    canvas.translate(screen.dx, screen.dy);
   177	
   178	    final magneticHeadingRad = (heading ?? 0) * (math.pi / 180);
   179	    final declinationRad = magneticDeclination * (math.pi / 180);
   180	    final trueHeadingRad = magneticHeadingRad + declinationRad;
   181	
   182	    final totalRotation =
   183	        trueHeadingRad + transformState.rotationRadians - mapRotation;
   184	
   185	    canvas.rotate(totalRotation);
   186	    canvas.drawPath(path, innerShadowPaint);
   187	    canvas.drawPath(path, outlinePaint);
   188	    canvas.restore();
   189	  }

   224	  void _drawLine(Canvas canvas, Offset from, Offset to) {
   225	    final paint = Paint()
   226	      ..color = Colors.red.withAlpha(180)
   227	      ..strokeWidth = 2.0
   228	      ..style = PaintingStyle.stroke

   234	  void _drawDashedLine(Canvas canvas, Offset from, Offset to, Paint paint) {
   235	    const dashLength = 10.0;
   236	    const gapLength = 6.0;
   237	    final totalLength = (to - from).distance;
   238	    if (totalLength < 1) return;
   239	
   240	    final direction = (to - from) / totalLength;
   241	    double drawn = 0;
   242	    bool drawing = true;
   243	
   244	    while (drawn < totalLength) {
   245	      final segLen = drawing ? dashLength : gapLength;
   246	      final end = math.min(drawn + segLen, totalLength);
   247	      if (drawing) {
   248	        canvas.drawLine(from + direction * drawn, from + direction * end, paint);
   249	      }
   250	      drawn = end;
   251	      drawing = !drawing;
   252	    }
   253	  }

   291	  bool shouldRepaint(covariant MapOverlayPainter oldDelegate) {
   292	    // Быстрые проверки простых полей
   293	    if (oldDelegate.transformState != transformState ||
   294	        oldDelegate.heading != heading ||
   295	        oldDelegate.mapRotation != mapRotation ||
   296	        oldDelegate.magneticDeclination != magneticDeclination ||
   297	        oldDelegate.previewDistanceMeters != previewDistanceMeters ||
   298	        oldDelegate.previewBearingDegrees != previewBearingDegrees ||
   299	        oldDelegate.currentUserImagePoint != currentUserImagePoint ||
   300	        oldDelegate.activeTargetImagePoint != activeTargetImagePoint) {
   301	      return true;
   302	    }
   303	
   304	    // Проверка списков: размер и последний элемент для userPath
   305	    if (oldDelegate.anchors.length != anchors.length) return true;
   306	    if (oldDelegate.targets.length != targets.length) return true;
   307	    if (oldDelegate.userPath.length != userPath.length) return true;
   308	    if (oldDelegate.pathJumpIndices.length != pathJumpIndices.length) return true;
   309	
   310	    // Если добавилась новая точка пути – перерисовать
   311	    if (userPath.isNotEmpty && oldDelegate.userPath.isNotEmpty) {
   312	      if (userPath.last != oldDelegate.userPath.last) return true;
   313	    } else if (userPath.isNotEmpty != oldDelegate.userPath.isNotEmpty) {
   314	      return true;
   315	    }
   316	
   317	    // Для якорей и целей – сравнение последних ID (обычно добавляются в конец)
   318	    if (anchors.isNotEmpty && oldDelegate.anchors.isNotEmpty) {
   319	      if (anchors.last.id != oldDelegate.anchors.last.id) return true;
   320	    } else if (anchors.isNotEmpty != oldDelegate.anchors.isNotEmpty) {
   321	      return true;
   322	    }
   323	
   324	    if (targets.isNotEmpty && oldDelegate.targets.isNotEmpty) {
   325	      if (targets.last.id != oldDelegate.targets.last.id) return true;
   326	    } else if (targets.isNotEmpty != oldDelegate.targets.isNotEmpty) {
   327	      return true;
   328	    }
   329	
   330	    return false;
   331	  }
