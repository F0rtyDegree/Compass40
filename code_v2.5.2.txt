    43	    }

    56	        currentGpsData.longitude!,

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) + math.cos(phi1) * math.sin(dR) * math.cos(theta);
    19	  final phi2 = math.asin(sinPhi2);
    20	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    21	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    22	  final lambda2 = lambda1 + math.atan2(y, x);
    23	
    24	  return {
    25	    'latitude': phi2 * 180 / math.pi,
    26	    'longitude': lambda2 * 180 / math.pi,
    27	  };
    28	}

    30	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    31	  const R = 6371e3;
    32	  final phi1 = lat1 * math.pi / 180;
    33	  final phi2 = lat2 * math.pi / 180;
    34	  final dPhi = (lat2 - lat1) * math.pi / 180;
    35	  final dLambda = (lon2 - lon1) * math.pi / 180;
    36	
    37	  final a = math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    38	      math.cos(phi1) * math.cos(phi2) *
    39	          math.sin(dLambda / 2) * math.sin(dLambda / 2);
    40	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    41	  return R * c;
    42	}
    43	
    44	/// Возвращает истинный азимут
    45	double calculateTrueBearing(double lat1, double lon1, double lat2, double lon2) {
    46	  final phi1 = lat1 * math.pi / 180;
    47	  final phi2 = lat2 * math.pi / 180;
    48	  final dLambda = (lon2 - lon1) * math.pi / 180;
    49	
    50	  final y = math.sin(dLambda) * math.cos(phi2);
    51	  final x =
    52	      math.cos(phi1) * math.sin(phi2) - math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    53	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    54	}
