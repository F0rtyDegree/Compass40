   172	  double _calculateCircularMedian(List<double> samples) {
   173	    if (samples.isEmpty) return 0.0;
   174	    if (samples.length == 1) return samples[0];
   175	
   176	    samples.sort();
   177	
   178	    double maxGap = 0;
   179	    int maxGapIndex = -1;
   180	
   181	    for (int i = 0; i < samples.length - 1; i++) {
   182	      final gap = samples[i + 1] - samples[i];
   183	      if (gap > maxGap) {
   184	        maxGap = gap;
   185	        maxGapIndex = i;
   186	      }
   187	    }
   188	
   189	    final wrapAroundGap = (samples.first + 360) - samples.last;
   190	    if (wrapAroundGap > maxGap) {
   191	      maxGap = wrapAroundGap;
   192	      maxGapIndex = samples.length - 1;
   193	    }
   194	
   195	    List<double> shiftedSamples;
   196	    if (maxGapIndex == samples.length - 1) {
   197	      shiftedSamples = List.from(samples);
   198	    } else {
   199	      shiftedSamples = [];
   200	      final shiftPoint = samples[maxGapIndex];
   201	      for (final s in samples) {
   202	        if (s > shiftPoint) {
   203	          shiftedSamples.add(s);
   204	        } else {
   205	          shiftedSamples.add(s + 360);
   206	        }
   207	      }
   208	    }
   209	
   210	    double median;
   211	    int mid = shiftedSamples.length ~/ 2;
   212	    if (shiftedSamples.length % 2 == 1) {
   213	      median = shiftedSamples[mid];
   214	    } else {
   215	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   216	    }
   217	
   218	    return median % 360;
   219	  }

   221	  void _updateHeading() {
   222	    final now = DateTime.now().millisecondsSinceEpoch;
   223	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   224	    final gpsBearing = state.gpsBearingNotifier.value;
   225	
   226	    bool useGps = false;
   227	
   228	    switch (state.compassMode) {
   229	      case CompassMode.magnetic:
   230	        useGps = false;
   231	        break;
   232	      case CompassMode.gps:
   233	        useGps = gpsBearing != null;
   234	        break;
   235	      case CompassMode.auto:
   236	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   237	        break;
   238	    }
   239	
   240	    state.isGpsCompassActiveNotifier.value = useGps;
   241	
   242	    double newHeading;
   243	
   244	    if (useGps) {
   245	      // GPS-компас
   246	      state.gpsBearingSamples.removeWhere(
   247	        (s) => now - s.$2 > state.averagingPeriod,
   248	      );
   249	      if (state.gpsBearingSamples.isEmpty) return;
   250	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   251	      final medianTrueBearing = _calculateCircularMedian(bearings);
   252	      // Преобразуем истинный курс в магнитный
   253	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   254	    } else {
   255	      // Магнитный компас
   256	      state.headingSamples.removeWhere(
   257	        (s) => now - s.$2 > state.averagingPeriod,
   258	      );
   259	      if (state.headingSamples.isEmpty) return;
   260	      final headings = state.headingSamples.map((s) => s.$1).toList();
   261	      // Данные с сенсора уже магнитные
   262	      newHeading = _calculateCircularMedian(headings);
   263	    }
   264	
   265	    // Сглаживание
   266	    double diff = newHeading - state.filteredHeading;
   267	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   268	
   269	    state.filteredHeading += state.smoothingFactor * diff;
   270	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   271	
   272	    state.headingNotifier.value = state.filteredHeading;
   273	  }

   279	  void _calculateWaypointData() {

   304	  Offset screenToImage(Offset screenPoint) {
   305	    if (state.imageSize == null || state.viewportSize == null) {
   306	      return screenPoint;
   307	    }
   308	
   309	    final vp = state.viewportSize!;
   310	    final t = state.transformState;
   311	    final imageSize = state.imageSize!;
   312	
   313	    final center = Offset(vp.width / 2, vp.height / 2);
   314	    final relative = screenPoint - center - t.translation;
   315	
   316	    final angle = -t.rotationRadians;
   317	    final cos = math.cos(angle);
   318	    final sin = math.sin(angle);
   319	    final derotated = Offset(
   320	      relative.dx * cos - relative.dy * sin,
   321	      relative.dx * sin + relative.dy * cos,
   322	    );
   323	
   324	    final unscaled = derotated / t.scale;
   325	    return unscaled + Offset(imageSize.width / 2, imageSize.height / 2);
   326	  }

   328	  Offset imageToScreen(Offset imagePoint) {
   329	    if (state.imageSize == null || state.viewportSize == null) {
   330	      return imagePoint;
   331	    }

   333	    final vp = state.viewportSize!;
   334	    final t = state.transformState;
   335	    final imageSize = state.imageSize!;
   336	
   337	    final center = Offset(vp.width / 2, vp.height / 2);
   338	    final local =
   339	        imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   340	    final scaled = local * t.scale;
   341	
   342	    final angle = t.rotationRadians;
   343	    final cos = math.cos(angle);
   344	    final sin = math.sin(angle);
   345	    final rotated = Offset(
   346	      scaled.dx * cos - scaled.dy * sin,
   347	      scaled.dx * sin + scaled.dy * cos,
   348	    );
   349	
   350	    return center + t.translation + rotated;

   714	  void _centerMapOnUser() {
   715	    final imagePoint = state.currentUserImagePoint;
   716	    if (imagePoint == null || state.viewportSize == null) return;
   717	
   718	    final crosshairScreen = _getCrosshairScreenPoint();
   719	    final vp = state.viewportSize!;
   720	    final vpCenter = Offset(vp.width / 2, vp.height / 2);
   721	    final targetTranslation = crosshairScreen - vpCenter;
   722	
   723	    final t = state.transformState;
   724	    final imageSize = state.imageSize;
   725	    if (imageSize == null) return;
   726	
   727	    final local =
   728	        imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   729	    final scaled = local * t.scale;
   730	
   731	    final angle = t.rotationRadians;
   732	    final cos = math.cos(angle);
   733	    final sin = math.sin(angle);
   734	    final rotated = Offset(
   735	      scaled.dx * cos - scaled.dy * sin,
   736	      scaled.dx * sin + scaled.dy * cos,
   737	    );
   738	
   739	    final newTranslation = targetTranslation - rotated;
   740	
   741	    setState(() {
   742	      state.transformState = MapTransformState(
   743	        scale: t.scale,
   744	        rotationRadians: t.rotationRadians,
   745	        translation: newTranslation,
   746	      );
   747	    });
   748	
   749	    _recalculateUserScreenPoint();
   750	  }

   373	  void _onScaleStart(ScaleStartDetails details) {
   374	    _gestureStartTranslation = _state.transformState.translation;
   375	    _gestureStartScale = _state.transformState.scale;
   376	    _gestureStartRotation = _state.transformState.rotationRadians;
   377	    _gestureStartFocalPoint = details.focalPoint;
   378	
   379	    // Запоминаем пиксель под прицелом
   380	    if (_state.viewportSize != null && _state.imageSize != null) {
   381	      final pivotScreen = _logic.getCrosshairScreenPoint();
   382	      _gestureStartPivotImage = _logic.screenToImage(pivotScreen);
   383	    } else {
   384	      _gestureStartPivotImage = null;
   385	    }
   386	
   387	    // Для вращения
   388	    _accumulatedRotation = 0.0;
   389	    if (_state.viewportSize != null && _state.imageSize != null) {
   390	      final pivotScreen = _logic.getCrosshairScreenPoint();
   391	      final startVector = _gestureStartFocalPoint - pivotScreen;
   392	      _lastAngle = math.atan2(startVector.dy, startVector.dx);
   393	    } else {
   394	      _lastAngle = 0.0;
   395	    }
   396	  }

   398	  void _onScaleUpdate(ScaleUpdateDetails details) {
   399	    if (_state.viewportSize == null || _state.imageSize == null) return;
   400	
   401	    final pointerCount = details.pointerCount;
   402	
   403	    // --------------------------------------------------------
   404	    // 1 палец
   405	    // --------------------------------------------------------
   406	    if (pointerCount == 1) {
   407	      if (_state.rotateMode) {
   408	        // Вращение одним пальцем вокруг прицела
   409	        final currentFocalPoint = details.focalPoint;
   410	        final pivotScreen = _logic.getCrosshairScreenPoint();
   411	
   412	        final currentVector = currentFocalPoint - pivotScreen;
   413	        final currentAngle = math.atan2(currentVector.dy, currentVector.dx);
   414	
   415	        double deltaAngle = currentAngle - _lastAngle;
   416	        if (deltaAngle > math.pi) deltaAngle -= 2 * math.pi;
   417	        if (deltaAngle < -math.pi) deltaAngle += 2 * math.pi;
   418	
   419	        _accumulatedRotation += deltaAngle;
   420	        _lastAngle = currentAngle;
   421	
   422	        const sensitivity = 0.8;
   423	        final newRotation = _gestureStartRotation + _accumulatedRotation * sensitivity;
   424	
   425	        if (_gestureStartPivotImage != null) {
   426	          final tempTransform = MapTransformState(
   427	            scale: _state.transformState.scale,
   428	            rotationRadians: newRotation,
   429	            translation: _gestureStartTranslation,
   430	          );
   431	
   432	          final oldTransform = _state.transformState;
   433	          _state.transformState = tempTransform;
   434	
   435	          final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   436	
   437	          _state.transformState = oldTransform;
   438	
   439	          final delta = pivotScreen - pivotScreenAfterRotate;
   440	          final newTranslation = _gestureStartTranslation + delta;
   441	
   442	          _logic.updateTransform(
   443	            MapTransformState(
   444	              scale: _state.transformState.scale,
   445	              rotationRadians: newRotation,
   446	              translation: newTranslation,
   447	            ),
   448	          );
   449	        } else {
   450	          _logic.updateTransform(
   451	            MapTransformState(
   452	              scale: _state.transformState.scale,
   453	              rotationRadians: newRotation,
   454	              translation: _state.transformState.translation,
   455	            ),
   456	          );
   457	        }
   458	      } else {
   459	        // Панорамирование
   460	        final delta = details.focalPoint - _gestureStartFocalPoint;
   461	        final newTranslation = _gestureStartTranslation + delta;
   462	
   463	        _logic.updateTransform(
   464	          MapTransformState(
   465	            scale: _state.transformState.scale,
   466	            rotationRadians: _state.transformState.rotationRadians,
   467	            translation: newTranslation,
   468	          ),
   469	        );
   470	      }
   471	      return;
   472	    }
   473	
   474	    // --------------------------------------------------------
   475	    // 2 пальца — умное масштабирование/вращение
   476	    // --------------------------------------------------------
   477	    if (pointerCount == 2) {
   478	      final scaleChange = (details.scale - 1.0).abs();
   479	      final rotationChange = details.rotation.abs();
   480	
   481	      // Определяем, что преобладает
   482	      if (scaleChange > rotationChange) {
   483	        // ---------- МАСШТАБИРОВАНИЕ ----------
   484	        final newScale = (_gestureStartScale * details.scale).clamp(0.05, 20.0);
   485	
   486	        if (_gestureStartPivotImage != null) {
   487	          final pivotScreen = _logic.getCrosshairScreenPoint();
   488	
   489	          final tempTransform = MapTransformState(
   490	            scale: newScale,
   491	            rotationRadians: _state.transformState.rotationRadians,
   492	            translation: _gestureStartTranslation,
   493	          );
   494	
   495	          final oldTransform = _state.transformState;
   496	          _state.transformState = tempTransform;
   497	
   498	          final pivotScreenAfterScale = _logic.imageToScreen(_gestureStartPivotImage!);
   499	
   500	          _state.transformState = oldTransform;
   501	
   502	          final delta = pivotScreen - pivotScreenAfterScale;
   503	          final newTranslation = _gestureStartTranslation + delta;
   504	
   505	          _logic.updateTransform(
   506	            MapTransformState(
   507	              scale: newScale,
   508	              rotationRadians: _state.transformState.rotationRadians,
   509	              translation: newTranslation,
   510	            ),
   511	          );
   512	        } else {
   513	          _logic.updateTransform(
   514	            MapTransformState(
   515	              scale: newScale,
   516	              rotationRadians: _state.transformState.rotationRadians,
   517	              translation: _gestureStartTranslation,
   518	            ),
   519	          );
   520	        }
   521	      } else {
   522	        // ---------- ВРАЩЕНИЕ ----------
   523	        final pivotScreen = _logic.getCrosshairScreenPoint();
   524	
   525	        // Накопление угла для плавного вращения
   526	        final currentRotation = _gestureStartRotation + details.rotation;
   527	
   528	        if (_gestureStartPivotImage != null) {
   529	          final tempTransform = MapTransformState(
   530	            scale: _state.transformState.scale,
   531	            rotationRadians: currentRotation,
   532	            translation: _gestureStartTranslation,
   533	          );
   534	
   535	          final oldTransform = _state.transformState;
   536	          _state.transformState = tempTransform;
   537	
   538	          final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   539	
   540	          _state.transformState = oldTransform;
   541	
   542	          final delta = pivotScreen - pivotScreenAfterRotate;
   543	          final newTranslation = _gestureStartTranslation + delta;
   544	
   545	          _logic.updateTransform(
   546	            MapTransformState(
   547	              scale: _state.transformState.scale,
   548	              rotationRadians: currentRotation,
   549	              translation: newTranslation,
   550	            ),
   551	          );
   552	        } else {
   553	          _logic.updateTransform(
   554	            MapTransformState(
   555	              scale: _state.transformState.scale,
   556	              rotationRadians: currentRotation,
   557	              translation: _state.transformState.translation,
   558	            ),
   559	          );
   560	        }
   561	      }
   562	      return;
   563	    }
   564	
   565	    // --------------------------------------------------------
   566	    // 3 и более пальцев — вращение вокруг прицела
   567	    // --------------------------------------------------------
   568	    if (pointerCount >= 3) {
   569	      final currentFocalPoint = details.focalPoint;
   570	      final pivotScreen = _logic.getCrosshairScreenPoint();
   571	
   572	      final startVector = _gestureStartFocalPoint - pivotScreen;
   573	      final currentVector = currentFocalPoint - pivotScreen;
   574	
   575	      final startAngle = math.atan2(startVector.dy, startVector.dx);
   576	      final currentAngle = math.atan2(currentVector.dy, currentVector.dx);
   577	
   578	      double deltaAngle = currentAngle - startAngle;
   579	      if (deltaAngle > math.pi) deltaAngle -= 2 * math.pi;
   580	      if (deltaAngle < -math.pi) deltaAngle += 2 * math.pi;
   581	
   582	      const sensitivity = 0.8;
   583	      final newRotation = _gestureStartRotation + deltaAngle * sensitivity;
   584	
   585	      if (_gestureStartPivotImage != null) {
   586	        final tempTransform = MapTransformState(
   587	          scale: _state.transformState.scale,
   588	          rotationRadians: newRotation,
   589	          translation: _gestureStartTranslation,
   590	        );
   591	
   592	        final oldTransform = _state.transformState;
   593	        _state.transformState = tempTransform;
   594	
   595	        final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   596	
   597	        _state.transformState = oldTransform;
   598	
   599	        final delta = pivotScreen - pivotScreenAfterRotate;
   600	        final newTranslation = _gestureStartTranslation + delta;
   601	
   602	        _logic.updateTransform(
   603	          MapTransformState(
   604	            scale: _state.transformState.scale,
   605	            rotationRadians: newRotation,
   606	            translation: newTranslation,
   607	          ),
   608	        );
   609	      } else {
   610	        _logic.updateTransform(
   611	          MapTransformState(
   612	            scale: _state.transformState.scale,
   613	            rotationRadians: newRotation,
   614	            translation: _state.transformState.translation,
   615	          ),
   616	        );
   617	      }
   618	      return;
   619	    }

    25	  // ---------------------------------------------------------
    26	  // Выбор рабочей пары
    27	  // ---------------------------------------------------------
    28	
    29	  /// Выбирает рабочую пару по правилу:
    30	  /// latest = последняя по времени создания точка
    31	  /// reference = ближайшая предыдущая, расстояние до которой >= 50 м
    32	  MapWorkingPair? selectWorkingPair(
    33	    List<MapAnchor> anchors, {
    34	    double minDistanceMeters = 50.0,
    35	  }) {
    36	    if (anchors.length < 2) return null;
    37	
    38	    final sorted = List<MapAnchor>.from(anchors)
    39	      ..sort((a, b) => a.createdAt.compareTo(b.createdAt));
    40	
    41	    final latest = sorted.last;
    42	
    43	    // Идём от предпоследней точки назад
    44	    for (int i = sorted.length - 2; i >= 0; i--) {
    45	      final candidate = sorted[i];
    46	      final dist = distanceBetweenAnchorsMeters(latest, candidate);
    47	      if (dist >= minDistanceMeters) {
    48	        return MapWorkingPair(latest: latest, reference: candidate);
    49	      }
    50	    }
    51	
    52	    return null;
    53	  }

    59	  double distanceBetweenAnchorsMeters(MapAnchor a, MapAnchor b) {
    60	    return _haversineDistance(a.latitude, a.longitude, b.latitude, b.longitude);
    61	  }

    63	  double _haversineDistance(
    64	    double lat1,
    65	    double lon1,
    66	    double lat2,
    67	    double lon2,
    68	  ) {
    69	    final phi1 = lat1 * math.pi / 180;
    70	    final phi2 = lat2 * math.pi / 180;
    71	    final dPhi = (lat2 - lat1) * math.pi / 180;
    72	    final dLambda = (lon2 - lon1) * math.pi / 180;
    73	
    74	    final a =
    75	        math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    76	        math.cos(phi1) *
    77	            math.cos(phi2) *
    78	            math.sin(dLambda / 2) *
    79	            math.sin(dLambda / 2);
    80	    final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    81	
    82	    return _earthRadius * c;
    83	  }

   131	    // Вектор в image-координатах
   132	    final imgVec = Offset(
   133	      pair.latest.imageX - pair.reference.imageX,
   134	      pair.latest.imageY - pair.reference.imageY,

   139	
   140	    // Опорная точка — reference.geo
   141	    final originLat = pair.reference.latitude;
   142	    final originLon = pair.reference.longitude;
   143	
   144	    // Вектор в локальных метрах
   145	    final latestLocal = geoToLocal(
   146	      latitude: pair.latest.latitude,

   157	
   158	    // Масштаб: метры на пиксель
   159	    final scale = geoLen / imgLen;
   160

   163	        math.atan2(geoVec.dy, geoVec.dx) - math.atan2(imgVec.dy, imgVec.dx);

   166	      originLat: originLat,
   167	      originLon: originLon,
   168	      referenceImageX: pair.reference.imageX,
   169	      referenceImageY: pair.reference.imageY,

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}
    30	
    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}
    47	
    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    79	  void _onScaleUpdate(ScaleUpdateDetails details, Size viewportSize) {
    80	    final center = Offset(viewportSize.width / 2, viewportSize.height / 2);
    81	
    82	    // Смещение фокальной точки с момента начала жеста
    83	    final focalDelta = details.focalPoint - _gestureStartFocalPoint;
    84	
    85	    // Новый масштаб
    86	    final newScale = (_gestureStartScale * details.scale).clamp(0.05, 20.0);
    87	
    88	    // Поворот: изменение угла с начала жеста
    89	    final deltaRotation = details.rotation - _previousGestureRotation;
    90	    _previousGestureRotation = details.rotation;
    91	    final newRotation = widget.transformState.rotationRadians + deltaRotation;
    92	
    93	    // Смещение: комбинация pan + сохранение центра масштаба/поворота
    94	    // Пересчитываем translation так, чтобы focal point оставался на месте
    95	    final scaleRatio = newScale / _gestureStartScale;
    96	
    97	    // Вектор от центра viewport до фокальной точки в начале жеста
    98	    final focalFromCenter = _gestureStartFocalPoint - center;

   100	    // После масштаба + поворота этот вектор увеличивается
   101	    final rotDelta = newRotation - _gestureStartRotation;
   102	    final cos = math.cos(rotDelta);
   103	    final sin = math.sin(rotDelta);

   105	    final rotatedFocal = Offset(
   106	      focalFromCenter.dx * cos - focalFromCenter.dy * sin,
   107	      focalFromCenter.dx * sin + focalFromCenter.dy * cos,
   108	    );

   111	        _gestureStartTranslation +
   112	        focalDelta -

   118	        rotationRadians: newRotation,

    59	class _ImagePainter extends CustomPainter {
    60	  final String imagePath;
    61	  final Size imageSize;
    62	  final double scale;
    63	  final double rotation;
    64	  final double centerX;
    65	  final double centerY;
    66	
    67	  // Статический кеш загруженных изображений
    68	  static final Map<String, ui.Image?> _cache = {};
    69	  static final Map<String, bool> _loading = {};
    70	
    71	  _ImagePainter({
    72	    required this.imagePath,
    73	    required this.imageSize,
    74	    required this.scale,
    75	    required this.rotation,
    76	    required this.centerX,
    77	    required this.centerY,
    78	  });
    79	
    80	  @override
    81	  void paint(Canvas canvas, Size size) {
    82	    final cachedImage = _cache[imagePath];
    83	
    84	    if (cachedImage == null) {
    85	      // Рисуем placeholder
    86	      _drawPlaceholder(canvas);
    87	      // Запускаем загрузку если ещё не загружается
    88	      if (_loading[imagePath] != true) {
    89	        _loading[imagePath] = true;
    90	        _loadImage(imagePath);
    91	      }
    92	      return;
    93	    }
    94	
    95	    canvas.save();
    96	
    97	    // Переходим в точку отображения центра карты
    98	    canvas.translate(centerX, centerY);
    99	    canvas.rotate(rotation);
   100	    canvas.scale(scale);
   101	
   102	    // Рисуем изображение со смещением так, чтобы его центр совпал с (0,0)
   103	    canvas.drawImage(
   104	      cachedImage,
   105	      Offset(-imageSize.width / 2, -imageSize.height / 2),
   106	      Paint()..filterQuality = FilterQuality.medium,
   107	    );
   108	
   109	    canvas.restore();
   110	  }
   111	
   112	  void _drawPlaceholder(Canvas canvas) {
   113	    final paint = Paint()
   114	      ..color = const Color(0xFFE0E0E0)
   115	      ..style = PaintingStyle.fill;
   116	    canvas.drawRect(
   117	      Rect.fromCenter(
   118	        center: Offset(centerX, centerY),
   119	        width: imageSize.width * scale,
   120	        height: imageSize.height * scale,
   121	      ),
   122	      paint,
   123	    );
   124	  }
   125	
   126	  Future<void> _loadImage(String path) async {
   127	    try {
   128	      final bytes = await File(path).readAsBytes();
   129	      final codec = await ui.instantiateImageCodec(bytes);
   130	      final frame = await codec.getNextFrame();
   131	      _cache[path] = frame.image;
   132	      _loading[path] = false;
   133	    } catch (e) {
   134	      _loading[path] = false;
   135	    }
   136	  }
   137	
   138	  @override
   139	  bool shouldRepaint(covariant _ImagePainter old) {
   140	    return old.scale != scale ||
   141	        old.rotation != rotation ||
   142	        old.centerX != centerX ||
   143	        old.centerY != centerY ||
   144	        old.imagePath != imagePath;
   145	  }

   222	  void _drawDashedLine(Canvas canvas, Offset from, Offset to, Paint paint) {
   223	    const dashLength = 10.0;
   224	    const gapLength = 6.0;
   225	    final totalLength = (to - from).distance;
   226	    if (totalLength < 1) return;
   227	
   228	    final direction = (to - from) / totalLength;
   229	    double drawn = 0;
   230	    bool drawing = true;
   231	
   232	    while (drawn < totalLength) {
   233	      final segLen = drawing ? dashLength : gapLength;
   234	      final end = math.min(drawn + segLen, totalLength);
   235	      if (drawing) {
   236	        canvas.drawLine(from + direction * drawn, from + direction * end, paint);
   237	      }
   238	      drawn = end;
   239	      drawing = !drawing;
   240	    }
   241	  }
