   173	  double _calculateCircularMedian(List<double> samples) {
   174	    if (samples.isEmpty) return 0.0;
   175	    if (samples.length == 1) return samples[0];
   176	
   177	    samples.sort();
   178	
   179	    double maxGap = 0;
   180	    int maxGapIndex = -1;
   181	
   182	    for (int i = 0; i < samples.length - 1; i++) {
   183	      final gap = samples[i + 1] - samples[i];
   184	      if (gap > maxGap) {
   185	        maxGap = gap;
   186	        maxGapIndex = i;
   187	      }
   188	    }
   189	
   190	    final wrapAroundGap = (samples.first + 360) - samples.last;
   191	    if (wrapAroundGap > maxGap) {
   192	      maxGap = wrapAroundGap;
   193	      maxGapIndex = samples.length - 1;
   194	    }
   195	
   196	    List<double> shiftedSamples;
   197	    if (maxGapIndex == samples.length - 1) {
   198	      shiftedSamples = List.from(samples);
   199	    } else {
   200	      shiftedSamples = [];
   201	      final shiftPoint = samples[maxGapIndex];
   202	      for (final s in samples) {
   203	        if (s > shiftPoint) {
   204	          shiftedSamples.add(s);
   205	        } else {
   206	          shiftedSamples.add(s + 360);
   207	        }
   208	      }
   209	    }
   210	
   211	    double median;
   212	    int mid = shiftedSamples.length ~/ 2;
   213	    if (shiftedSamples.length % 2 == 1) {
   214	      median = shiftedSamples[mid];
   215	    } else {
   216	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   217	    }
   218	
   219	    return median % 360;
   220	  }

   222	  void _updateHeading() {
   223	    final now = DateTime.now().millisecondsSinceEpoch;
   224	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   225	    final gpsBearing = state.gpsBearingNotifier.value;
   226	
   227	    bool useGps = false;
   228	
   229	    switch (state.compassMode) {
   230	      case CompassMode.magnetic:
   231	        useGps = false;
   232	        break;
   233	      case CompassMode.gps:
   234	        useGps = gpsBearing != null;
   235	        break;
   236	      case CompassMode.auto:
   237	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   238	        break;
   239	    }
   240	
   241	    state.isGpsCompassActiveNotifier.value = useGps;
   242	
   243	    double newHeading;
   244	
   245	    if (useGps) {
   246	      // GPS-компас
   247	      state.gpsBearingSamples.removeWhere(
   248	        (s) => now - s.$2 > state.averagingPeriod,
   249	      );
   250	      if (state.gpsBearingSamples.isEmpty) return;
   251	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   252	      final medianTrueBearing = _calculateCircularMedian(bearings);
   253	      // Преобразуем истинный курс в магнитный
   254	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   255	    } else {
   256	      // Магнитный компас
   257	      state.headingSamples.removeWhere(
   258	        (s) => now - s.$2 > state.averagingPeriod,
   259	      );
   260	      if (state.headingSamples.isEmpty) return;
   261	      final headings = state.headingSamples.map((s) => s.$1).toList();
   262	      // Данные с сенсора уже магнитные
   263	      newHeading = _calculateCircularMedian(headings);
   264	    }
   265	
   266	    // Сглаживание
   267	    double diff = newHeading - state.filteredHeading;
   268	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   269	
   270	    state.filteredHeading += state.smoothingFactor * diff;
   271	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   272	
   273	    state.headingNotifier.value = state.filteredHeading;
   274	  }

   255	    if (state.imageSize == null || state.viewportSize == null) {
   256	      return screenPoint;
   257	    }

   259	    final vp = state.viewportSize!;
   260	    final t = state.transformState;
   261	    final imageSize = state.imageSize!;
   262	
   263	    final center = Offset(vp.width / 2, vp.height / 2);
   264	
   265	    // Убираем перенос в центр и translation
   266	    final relative = screenPoint - center - t.translation;
   267	
   268	    // Обратный поворот
   269	    final angle = -t.rotationRadians;
   270	    final cos = math.cos(angle);
   271	    final sin = math.sin(angle);
   272	    final derotated = Offset(
   273	      relative.dx * cos - relative.dy * sin,
   274	      relative.dx * sin + relative.dy * cos,
   275	    );
   276	
   277	    // Обратный масштаб
   278	    final unscaled = derotated / t.scale;
   279	
   280	    // Возвращаем к центру изображения
   281	    return unscaled + Offset(imageSize.width / 2, imageSize.height / 2);
   282	  }

   296	    // Вектор от центра изображения
   297	    final local = imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   298	
   299	    // Масштаб
   300	    final scaled = local * t.scale;
   301	
   302	    // Поворот
   303	    final angle = t.rotationRadians;
   304	    final cos = math.cos(angle);
   305	    final sin = math.sin(angle);
   306	    final rotated = Offset(
   307	      scaled.dx * cos - scaled.dy * sin,
   308	      scaled.dx * sin + scaled.dy * cos,
   309	    );
   310	
   311	    return center + t.translation + rotated;

   650	  void _centerMapOnUser() {
   651	    final imagePoint = state.currentUserImagePoint;
   652	    if (imagePoint == null || state.viewportSize == null) return;
   653	
   654	    final vp = state.viewportSize!;
   655	    final crosshairScreen = _getCrosshairScreenPoint();
   656	    final targetCenter = crosshairScreen - Offset(vp.width / 2, vp.height / 2);
   657	
   658	    // Вычисляем translation, при котором imagePoint окажется под прицелом
   659	    final t = state.transformState;
   660	    final angle = t.rotationRadians;
   661	    final cos = math.cos(angle);
   662	    final sin = math.sin(angle);
   663	
   664	    if (state.imageSize == null) return;
   665	    final imageSize = state.imageSize!;
   666	
   667	    final local = imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   668	    final scaled = local * t.scale;
   669	    final rotated = Offset(
   670	      scaled.dx * cos - scaled.dy * sin,
   671	      scaled.dx * sin + scaled.dy * cos,
   672	    );
   673	
   674	    final newTranslation = targetCenter - rotated;
   675	
   676	    setState(() {
   677	      state.transformState = MapTransformState(
   678	        scale: t.scale,
   679	        rotationRadians: t.rotationRadians,
   680	        translation: newTranslation,
   681	      );
   682	    });

    29	  /// Выбирает рабочую пару по правилу:
    30	  /// latest = последняя по времени создания точка
    31	  /// reference = ближайшая предыдущая, расстояние до которой >= 50 м
    32	  MapWorkingPair? selectWorkingPair(
    33	    List<MapAnchor> anchors, {
    34	    double minDistanceMeters = 50.0,
    35	  }) {
    36	    if (anchors.length < 2) return null;
    37	
    38	    final sorted = List<MapAnchor>.from(anchors)
    39	      ..sort((a, b) => a.createdAt.compareTo(b.createdAt));
    40	
    41	    final latest = sorted.last;
    42	
    43	    // Идём от предпоследней точки назад
    44	    for (int i = sorted.length - 2; i >= 0; i--) {
    45	      final candidate = sorted[i];
    46	      final dist = distanceBetweenAnchorsMeters(latest, candidate);
    47	      if (dist >= minDistanceMeters) {
    48	        return MapWorkingPair(latest: latest, reference: candidate);
    49	      }
    50	    }
    51	
    52	    return null;
    53	  }

    59	  double distanceBetweenAnchorsMeters(MapAnchor a, MapAnchor b) {
    60	    return _haversineDistance(a.latitude, a.longitude, b.latitude, b.longitude);

    63	  double _haversineDistance(
    64	    double lat1,
    65	    double lon1,
    66	    double lat2,
    67	    double lon2,
    68	  ) {
    69	    final phi1 = lat1 * math.pi / 180;
    70	    final phi2 = lat2 * math.pi / 180;
    71	    final dPhi = (lat2 - lat1) * math.pi / 180;
    72	    final dLambda = (lon2 - lon1) * math.pi / 180;
    73	
    74	    final a = math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    75	        math.cos(phi1) *
    76	            math.cos(phi2) *
    77	            math.sin(dLambda / 2) *
    78	            math.sin(dLambda / 2);
    79	    final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    80	
    81	    return _earthRadius * c;
    82	  }

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}
    30	
    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}
    47	
    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    83	    final center = Offset(viewportSize.width / 2, viewportSize.height / 2);
    84	
    85	    // Смещение фокальной точки с момента начала жеста

   229	  void _drawDashedLine(Canvas canvas, Offset from, Offset to, Paint paint) {
   230	    const dashLength = 10.0;
   231	    const gapLength = 6.0;
   232	
   233	    final totalLength = (to - from).distance;
   234	    if (totalLength < 1) return;
   235	
   236	    final direction = (to - from) / totalLength;
   237	    double drawn = 0;
   238	    bool drawing = true;
   239	
   240	    while (drawn < totalLength) {
   241	      final segLen =
   242	          drawing ? dashLength : gapLength;
   243	      final end = math.min(drawn + segLen, totalLength);
   244	
   245	      if (drawing) {

   247	          from + direction * drawn,
   248	          from + direction * end,
   249	          paint,
   250	        );
   251	      }
   252	
   253	      drawn = end;
   254	      drawing = !drawing;
   255	    }
   256	  }
