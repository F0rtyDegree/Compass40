   172	  double _calculateCircularMedian(List<double> samples) {
   173	    if (samples.isEmpty) return 0.0;
   174	    if (samples.length == 1) return samples[0];
   175	
   176	    samples.sort();
   177	
   178	    double maxGap = 0;
   179	    int maxGapIndex = -1;
   180	
   181	    for (int i = 0; i < samples.length - 1; i++) {
   182	      final gap = samples[i + 1] - samples[i];
   183	      if (gap > maxGap) {
   184	        maxGap = gap;
   185	        maxGapIndex = i;
   186	      }
   187	    }
   188	
   189	    final wrapAroundGap = (samples.first + 360) - samples.last;
   190	    if (wrapAroundGap > maxGap) {
   191	      maxGap = wrapAroundGap;
   192	      maxGapIndex = samples.length - 1;
   193	    }
   194	
   195	    List<double> shiftedSamples;
   196	    if (maxGapIndex == samples.length - 1) {
   197	      shiftedSamples = List.from(samples);
   198	    } else {
   199	      shiftedSamples = [];
   200	      final shiftPoint = samples[maxGapIndex];
   201	      for (final s in samples) {
   202	        if (s > shiftPoint) {
   203	          shiftedSamples.add(s);
   204	        } else {
   205	          shiftedSamples.add(s + 360);
   206	        }
   207	      }
   208	    }
   209	
   210	    double median;
   211	    int mid = shiftedSamples.length ~/ 2;
   212	    if (shiftedSamples.length % 2 == 1) {
   213	      median = shiftedSamples[mid];
   214	    } else {
   215	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   216	    }
   217	
   218	    return median % 360;
   219	  }

   221	  void _updateHeading() {
   222	    final now = DateTime.now().millisecondsSinceEpoch;
   223	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   224	    final gpsBearing = state.gpsBearingNotifier.value;
   225	
   226	    bool useGps = false;
   227	
   228	    switch (state.compassMode) {
   229	      case CompassMode.magnetic:
   230	        useGps = false;
   231	        break;
   232	      case CompassMode.gps:
   233	        useGps = gpsBearing != null;
   234	        break;
   235	      case CompassMode.auto:
   236	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   237	        break;
   238	    }
   239	
   240	    state.isGpsCompassActiveNotifier.value = useGps;
   241	
   242	    double newHeading;
   243	
   244	    if (useGps) {
   245	      // GPS-компас
   246	      state.gpsBearingSamples.removeWhere(
   247	        (s) => now - s.$2 > state.averagingPeriod,
   248	      );
   249	      if (state.gpsBearingSamples.isEmpty) return;
   250	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   251	      final medianTrueBearing = _calculateCircularMedian(bearings);
   252	      // Преобразуем истинный курс в магнитный
   253	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   254	    } else {
   255	      // Магнитный компас
   256	      state.headingSamples.removeWhere(
   257	        (s) => now - s.$2 > state.averagingPeriod,
   258	      );
   259	      if (state.headingSamples.isEmpty) return;
   260	      final headings = state.headingSamples.map((s) => s.$1).toList();
   261	      // Данные с сенсора уже магнитные
   262	      newHeading = _calculateCircularMedian(headings);
   263	    }
   264	
   265	    // Сглаживание
   266	    double diff = newHeading - state.filteredHeading;
   267	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   268	
   269	    state.filteredHeading += state.smoothingFactor * diff;
   270	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   271	
   272	    state.headingNotifier.value = state.filteredHeading;
   273	  }

   279	  void _calculateWaypointData() {

   670	  void _centerMapOnUser() {
   671	    final imagePoint = state.currentUserImagePoint;
   672	    if (imagePoint == null || state.viewportSize == null) return;
   673	
   674	    final crosshairScreen = _getCrosshairScreenPoint();
   675	    final vp = state.viewportSize!;
   676	    final vpCenter = Offset(vp.width / 2, vp.height / 2);
   677	    final targetTranslation = crosshairScreen - vpCenter;
   678	
   679	    final t = state.transformState;
   680	    final imageSize = state.imageSize;
   681	    if (imageSize == null) return;
   682	
   683	    final local =
   684	        imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   685	    final scaled = local * t.scale;
   686	
   687	    final angle = t.rotationRadians;
   688	    final cos = math.cos(angle);
   689	    final sin = math.sin(angle);
   690	    final rotated = Offset(
   691	      scaled.dx * cos - scaled.dy * sin,
   692	      scaled.dx * sin + scaled.dy * cos,
   693	    );
   694	
   695	    final newTranslation = targetTranslation - rotated;
   696	
   697	    setState(() {
   698	      state.transformState = MapTransformState(
   699	        scale: t.scale,
   700	        rotationRadians: t.rotationRadians,
   701	        translation: newTranslation,
   702	      );
   703	    });
   704	
   705	    _recalculateUserScreenPoint();
   706	  }

   399

   414	        if (deltaAngle < -math.pi) deltaAngle += 2 * math.pi;

    22	class MapCalibrationService {
    23	  static const double _earthRadius = 6371000.0;
    24	
    25	  // ---------------------------------------------------------
    26	  // Выбор рабочей пары
    27	  // ---------------------------------------------------------
    28	
    29	  /// Выбирает рабочую пару по правилу:
    30	  /// latest = последняя по времени создания точка
    31	  /// reference = ближайшая предыдущая, расстояние до которой >= 50 м
    32	  MapWorkingPair? selectWorkingPair(
    33	    List<MapAnchor> anchors, {
    34	    double minDistanceMeters = 50.0,
    35	  }) {
    36	    if (anchors.length < 2) return null;
    37	
    38	    final sorted = List<MapAnchor>.from(anchors)
    39	      ..sort((a, b) => a.createdAt.compareTo(b.createdAt));
    40	
    41	    final latest = sorted.last;
    42	
    43	    // Идём от предпоследней точки назад
    44	    for (int i = sorted.length - 2; i >= 0; i--) {
    45	      final candidate = sorted[i];
    46	      final dist = distanceBetweenAnchorsMeters(latest, candidate);
    47	      if (dist >= minDistanceMeters) {
    48	        return MapWorkingPair(latest: latest, reference: candidate);
    49	      }
    50	    }
    51	
    52	    return null;
    53	  }
    54	
    55	  // ---------------------------------------------------------
    56	  // Расстояние между точками привязки
    57	  // ---------------------------------------------------------
    58	
    59	  double distanceBetweenAnchorsMeters(MapAnchor a, MapAnchor b) {
    60	    return _haversineDistance(a.latitude, a.longitude, b.latitude, b.longitude);
    61	  }
    62	
    63	  double _haversineDistance(
    64	    double lat1,
    65	    double lon1,
    66	    double lat2,
    67	    double lon2,
    68	  ) {
    69	    final phi1 = lat1 * math.pi / 180;
    70	    final phi2 = lat2 * math.pi / 180;
    71	    final dPhi = (lat2 - lat1) * math.pi / 180;
    72	    final dLambda = (lon2 - lon1) * math.pi / 180;
    73	
    74	    final a =
    75	        math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    76	        math.cos(phi1) *
    77	            math.cos(phi2) *
    78	            math.sin(dLambda / 2) *
    79	            math.sin(dLambda / 2);
    80	    final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    81	
    82	    return _earthRadius * c;
    83	  }
    84	
    85	  // ---------------------------------------------------------
    86	  // Перевод GPS → LocalPoint (метры East/North)
    87	  // ---------------------------------------------------------
    88	
    89	  /// Переводит GPS-точку в локальные метры
    90	  /// относительно опорной точки (origin)
    91	  LocalPoint geoToLocal({
    92	    required double latitude,
    93	    required double longitude,
    94	    required double originLat,
    95	    required double originLon,
    96	  }) {
    97	    final dLat = (latitude - originLat) * math.pi / 180;
    98	    final dLon = (longitude - originLon) * math.pi / 180;
    99	    final originLatRad = originLat * math.pi / 180;
   100	
   101	    final north = dLat * _earthRadius;
   102	    final east = dLon * _earthRadius * math.cos(originLatRad);
   103	
   104	    return LocalPoint(east: east, north: north);
   105	  }
   106	
   107	  /// Переводит локальные метры (East/North) обратно в GPS
   108	  GeoPoint localToGeo({
   109	    required LocalPoint local,
   110	    required double originLat,
   111	    required double originLon,
   112	  }) {
   113	    final originLatRad = originLat * math.pi / 180;
   114	
   115	    final dLat = local.north / _earthRadius;
   116	    final dLon = local.east / (_earthRadius * math.cos(originLatRad));
   117	
   118	    return GeoPoint(
   119	      latitude: originLat + dLat * 180 / math.pi,
   120	      longitude: originLon + dLon * 180 / math.pi,
   121	    );
   122	  }
   123	
   124	  // ---------------------------------------------------------
   125	  // Построение локального 2-точечного преобразования
   126	  // ---------------------------------------------------------
   127	
   128	  /// Вычисляет параметры similarity transform по двум точкам привязки.
   129	  /// Это ядро для image ↔ geo.
   130	  _SimilarityTransform? _buildTransform(MapWorkingPair pair) {
   131	    // Вектор в image-координатах
   132	    final imgVec = Offset(
   133	      pair.latest.imageX - pair.reference.imageX,
   134	      pair.latest.imageY - pair.reference.imageY,
   135	    );
   136	
   137	    final imgLen = imgVec.distance;
   138	    if (imgLen < 1e-9) return null; // Две точки совпали по пикселям
   139	
   140	    // Опорная точка — reference.geo
   141	    final originLat = pair.reference.latitude;
   142	    final originLon = pair.reference.longitude;
   143	
   144	    // Вектор в локальных метрах
   145	    final latestLocal = geoToLocal(
   146	      latitude: pair.latest.latitude,
   147	      longitude: pair.latest.longitude,
   148	      originLat: originLat,
   149	      originLon: originLon,
   150	    );
   151	
   152	    final geoVec = Offset(latestLocal.east, -latestLocal.north);
   153	    // Минус north потому что ось Y экрана направлена вниз, north — вверх
   154	
   155	    final geoLen = geoVec.distance;
   156	    if (geoLen < 1e-9) return null; // Две точки совпали по GPS
   157	
   158	    // Масштаб: метры на пиксель
   159	    final scale = geoLen / imgLen;
   160	
   161	    // Угол поворота: image -> geo
   162	    final angle =
   163	        math.atan2(geoVec.dy, geoVec.dx) - math.atan2(imgVec.dy, imgVec.dx);
   164	
   165	    return _SimilarityTransform(
   166	      originLat: originLat,
   167	      originLon: originLon,
   168	      referenceImageX: pair.reference.imageX,
   169	      referenceImageY: pair.reference.imageY,
   170	      scale: scale,
   171	      angleRadians: angle,
   172	    );
   173	  }
   174	
   175	  // ---------------------------------------------------------
   176	  // image → GPS
   177	  // ---------------------------------------------------------
   178	
   179	  /// Переводит пиксель изображения в GPS-координаты.
   180	  /// Требует рабочую пару.
   181	  GeoPoint? imagePointToGeo({
   182	    required Offset imagePoint,
   183	    required MapWorkingPair pair,
   184	  }) {
   185	    final t = _buildTransform(pair);
   186	    if (t == null) return null;
   187	
   188	    // Вектор относительно reference в image-пространстве
   189	    final dx = imagePoint.dx - t.referenceImageX;
   190	    final dy = imagePoint.dy - t.referenceImageY;
   191	
   192	    // Масштаб
   193	    final scaledDx = dx * t.scale;
   194	    final scaledDy = dy * t.scale;
   195	
   196	    // Поворот
   197	    final cos = math.cos(t.angleRadians);
   198	    final sin = math.sin(t.angleRadians);
   199	    final localEast = scaledDx * cos - scaledDy * sin;
   200	    final localNorthNeg = scaledDx * sin + scaledDy * cos;
   201	
   202	    // Обратно north (была инвертирована ось Y)
   203	    final localNorth = -localNorthNeg;
   204	
   205	    return localToGeo(
   206	      local: LocalPoint(east: localEast, north: localNorth),
   207	      originLat: t.originLat,
   208	      originLon: t.originLon,
   209	    );
   210	  }
   211	
   212	  // ---------------------------------------------------------
   213	  // GPS → image
   214	  // ---------------------------------------------------------
   215	
   216	  /// Переводит GPS-координаты в пиксель изображения.
   217	  /// Требует рабочую пару.
   218	  Offset? geoToImagePoint({
   219	    required double latitude,
   220	    required double longitude,
   221	    required MapWorkingPair pair,
   222	  }) {
   223	    final t = _buildTransform(pair);
   224	    if (t == null) return null;
   225	
   226	    // Локальные метры относительно reference
   227	    final local = geoToLocal(
   228	      latitude: latitude,
   229	      longitude: longitude,
   230	      originLat: t.originLat,
   231	      originLon: t.originLon,
   232	    );
   233	
   234	    // Инвертируем north → ось Y экрана
   235	    final geoX = local.east;
   236	    final geoY = -local.north;
   237	
   238	    // Обратный поворот
   239	    final cos = math.cos(-t.angleRadians);
   240	    final sin = math.sin(-t.angleRadians);
   241	    final rotX = geoX * cos - geoY * sin;
   242	    final rotY = geoX * sin + geoY * cos;
   243	
   244	    // Обратный масштаб
   245	    final dx = rotX / t.scale;
   246	    final dy = rotY / t.scale;
   247	
   248	    return Offset(t.referenceImageX + dx, t.referenceImageY + dy);
   249	  }

   260	  /// Расстояние и азимут от одной GPS-точки до другой
   261	  BearingAndDistance bearingAndDistance({
   262	    required double fromLat,
   263	    required double fromLon,
   264	    required double toLat,
   265	    required double toLon,
   266	    required double magneticDeclination,
   267	  }) {
   268	    final dist = _haversineDistance(fromLat, fromLon, toLat, toLon);
   269	
   270	    final phi1 = fromLat * math.pi / 180;
   271	    final phi2 = toLat * math.pi / 180;
   272	    final dLambda = (toLon - fromLon) * math.pi / 180;
   273	
   274	    final y = math.sin(dLambda) * math.cos(phi2);
   275	    final x =
   276	        math.cos(phi1) * math.sin(phi2) -
   277	        math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
   278	
   279	    final trueBearing = (math.atan2(y, x) * 180 / math.pi + 360) % 360;
   280	    final magneticBearing = (trueBearing - magneticDeclination + 360) % 360;
   281	
   282	    return BearingAndDistance(
   283	      distanceMeters: dist,
   284	      magneticBearing: magneticBearing,
   285	    );
   286	  }

     1	import 'dart:math' as math;

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}
    30	
    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}
    47	
    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    50	class WindRosePainter extends CustomPainter {
    51	  final bool isDarkMode;
    52	  final double heading;
    53	  WindRosePainter({required this.isDarkMode, required this.heading});
    54	
    55	  @override
    56	  void paint(Canvas canvas, Size size) {
    57	    const stroke = 22.0;
    58	    final radius = size.width / 2 - stroke / 2;
    59	    final center = Offset(size.width / 2, size.height / 2);
    60	    final colors = [
    61	      Colors.red,
    62	      Colors.green,
    63	      Colors.yellow,
    64	      Colors.cyan,
    65	      Colors.red,
    66	      Colors.green,
    67	      Colors.yellow,
    68	      Colors.cyan,
    69	    ];
    70	    final paint = Paint()
    71	      ..style = PaintingStyle.stroke
    72	      ..strokeWidth = stroke;
    73	
    74	    const sect = math.pi / 4;
    75	    for (int i = 0; i < 8; i++) {
    76	      paint.color = colors[i];
    77	      final start = -math.pi / 2 - sect / 2 + i * sect;
    78	      canvas.drawArc(
    79	        Rect.fromCircle(center: center, radius: radius),
    80	        start,
    81	        sect,
    82	        false,
    83	        paint,
    84	      );
    85	    }
    86	
    87	    // Разделительные линии (15 градусов)
    88	    final linePaint = Paint()
    89	      ..style = PaintingStyle.stroke
    90	      ..strokeWidth = 1.0;
    91	
    92	    const subSect = math.pi / 12; // 15 градусов
    93	    for (int i = 0; i < 8; i++) {
    94	      final sectorColor = colors[i];
    95	      final lineLuminance = sectorColor.computeLuminance();
    96	      linePaint.color = lineLuminance > 0.5
    97	          ? Colors.black.withAlpha(128)
    98	          : Colors.white.withAlpha(128);
    99	
   100	      final majorStart = -math.pi / 2 - sect / 2 + i * sect;
   101	      final line1Angle = majorStart + subSect;
   102	      final line2Angle = majorStart + 2 * subSect;
   103	
   104	      void drawLineAtAngle(double angle) {
   105	        final startPoint = Offset(
   106	          center.dx + (radius - stroke / 2) * math.cos(angle),
   107	          center.dy + (radius - stroke / 2) * math.sin(angle),
   108	        );
   109	        final endPoint = Offset(
   110	          center.dx + (radius + stroke / 2) * math.cos(angle),
   111	          center.dy + (radius + stroke / 2) * math.sin(angle),
   112	        );
   113	        canvas.drawLine(startPoint, endPoint, linePaint);
   114	      }
   115	
   116	      drawLineAtAngle(line1Angle);
   117	      drawLineAtAngle(line2Angle);
   118	    }
   119	
   120	    // координатные метки N,E,S,W
   121	    final letters = {0: 'С', 2: 'В', 4: 'Ю', 6: 'З'};
   122	    for (var e in letters.entries) {
   123	      final i = e.key;
   124	      final t = e.value;
   125	      final col = colors[i];
   126	      final textColor = col.computeLuminance() > 0.5
   127	          ? Colors.black
   128	          : Colors.white;
   129	      final tp = TextPainter(
   130	        text: TextSpan(
   131	          text: t,
   132	          style: TextStyle(
   133	            color: textColor,
   134	            fontWeight: FontWeight.bold,
   135	            fontSize: 14,
   136	          ),
   137	        ),
   138	        textDirection: TextDirection.ltr,
   139	      )..layout();
   140	      final ang = -math.pi / 2 + i * sect;
   141	      canvas.save();
   142	      canvas.translate(center.dx, center.dy);
   143	      canvas.rotate(ang);
   144	      canvas.translate(radius, 0);
   145	      canvas.rotate(-ang + heading * math.pi / 180);
   146	      tp.paint(canvas, Offset(-tp.width / 2, -tp.height / 2));
   147	      canvas.restore();
   148	    }
   149	  }
   150	
   151	  @override
   152	  bool shouldRepaint(covariant WindRosePainter old) =>
   153	      old.isDarkMode != isDarkMode || old.heading != heading;
   154	}

   233	  void _drawDashedLine(Canvas canvas, Offset from, Offset to, Paint paint) {
   234	    const dashLength = 10.0;
   235	    const gapLength = 6.0;
   236	
   237	    final totalLength = (to - from).distance;
   238	    if (totalLength < 1) return;
   239	
   240	    final direction = (to - from) / totalLength;
   241	    double drawn = 0;
   242	    bool drawing = true;
   243	
   244	    while (drawn < totalLength) {
   245	      final segLen = drawing ? dashLength : gapLength;
   246	      final end = math.min(drawn + segLen, totalLength);
   247	
   248	      if (drawing) {
   249	        canvas.drawLine(
   250	          from + direction * drawn,
   251	          from + direction * end,
   252	          paint,
   253	        );
   254	      }
   255	
   256	      drawn = end;
   257	      drawing = !drawing;
   258	    }
   259	  }
