   173	  double _calculateCircularMedian(List<double> samples) {
   174	    if (samples.isEmpty) return 0.0;
   175	    if (samples.length == 1) return samples[0];
   176	
   177	    samples.sort();
   178	
   179	    double maxGap = 0;
   180	    int maxGapIndex = -1;
   181	
   182	    for (int i = 0; i < samples.length - 1; i++) {
   183	      final gap = samples[i + 1] - samples[i];
   184	      if (gap > maxGap) {
   185	        maxGap = gap;
   186	        maxGapIndex = i;
   187	      }
   188	    }
   189	
   190	    final wrapAroundGap = (samples.first + 360) - samples.last;
   191	    if (wrapAroundGap > maxGap) {
   192	      maxGap = wrapAroundGap;
   193	      maxGapIndex = samples.length - 1;
   194	    }
   195	
   196	    List<double> shiftedSamples;
   197	    if (maxGapIndex == samples.length - 1) {
   198	      shiftedSamples = List.from(samples);
   199	    } else {
   200	      shiftedSamples = [];
   201	      final shiftPoint = samples[maxGapIndex];
   202	      for (final s in samples) {
   203	        if (s > shiftPoint) {
   204	          shiftedSamples.add(s);
   205	        } else {
   206	          shiftedSamples.add(s + 360);
   207	        }
   208	      }
   209	    }
   210	    
   211	    double median;
   212	    int mid = shiftedSamples.length ~/ 2;
   213	    if (shiftedSamples.length % 2 == 1) {
   214	      median = shiftedSamples[mid];
   215	    } else {
   216	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   217	    }
   218	
   219	    return median % 360;
   220	  }

   222	  void _updateHeading() {
   223	    final now = DateTime.now().millisecondsSinceEpoch;
   224	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   225	    final gpsBearing = state.gpsBearingNotifier.value;
   226	
   227	    bool useGps = false;
   228	
   229	    switch (state.compassMode) {
   230	      case CompassMode.magnetic:
   231	        useGps = false;
   232	        break;
   233	      case CompassMode.gps:
   234	        useGps = gpsBearing != null;
   235	        break;
   236	      case CompassMode.auto:
   237	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   238	        break;
   239	    }
   240	
   241	    state.isGpsCompassActiveNotifier.value = useGps;
   242	    
   243	    double newHeading;
   244	
   245	    if (useGps) {
   246	      // GPS-компас
   247	      state.gpsBearingSamples.removeWhere((s) => now - s.$2 > state.averagingPeriod);
   248	      if (state.gpsBearingSamples.isEmpty) return;
   249	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   250	      final medianTrueBearing = _calculateCircularMedian(bearings);
   251	      // Преобразуем истинный курс в магнитный
   252	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   253	
   254	    } else {
   255	      // Магнитный компас
   256	      state.headingSamples.removeWhere((s) => now - s.$2 > state.averagingPeriod);
   257	      if (state.headingSamples.isEmpty) return;
   258	      final headings = state.headingSamples.map((s) => s.$1).toList();
   259	      // Данные с сенсора уже магнитные
   260	      newHeading = _calculateCircularMedian(headings);
   261	    }
   262	    
   263	    // Сглаживание
   264	    double diff = newHeading - state.filteredHeading;
   265	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   266	
   267	    state.filteredHeading += state.smoothingFactor * diff;
   268	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   269	
   270	    state.headingNotifier.value = state.filteredHeading;
   271	  }

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}

    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}
    47	
    48	/// Возвращает истинный азимут
    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

   102	      final line2Angle = majorStart + 2 * subSect;

   107	          center.dy + (radius - stroke / 2) * math.sin(angle),

   111	          center.dy + (radius + stroke / 2) * math.sin(angle),

   117	      drawLineAtAngle(line2Angle);

   145	        val success = SensorManager.getRotationMatrix(
   146	            rotationMatrix,
   147	            null,
   148	            accelerometerReading,
   149	            magnetometerReading
   150	        )

   157	    private fun updateOrientationAnglesFromMatrix(sourceMatrix: FloatArray) {
   158	        val rotation = getDisplayRotation()
   159	
   160	        val (axisX, axisY) = when (rotation) {
   161	            Surface.ROTATION_90 -> Pair(SensorManager.AXIS_Y, SensorManager.AXIS_MINUS_X)
   162	            Surface.ROTATION_180 -> Pair(SensorManager.AXIS_MINUS_X, SensorManager.AXIS_MINUS_Y)
   163	            Surface.ROTATION_270 -> Pair(SensorManager.AXIS_MINUS_Y, SensorManager.AXIS_X)
   164	            else -> Pair(SensorManager.AXIS_X, SensorManager.AXIS_Y)
   165	        }
   166	
   167	        SensorManager.remapCoordinateSystem(
   168	            sourceMatrix,
   169	            axisX,
   170	            axisY,
   171	            remappedRotationMatrix
   172	        )
   173	
   174	        SensorManager.getOrientation(remappedRotationMatrix, orientationAngles)
   175	
   176	        val azimuthRad = orientationAngles[0].toDouble()
   177	        val azimuthDeg = Math.toDegrees(azimuthRad)
   178	
   179	        val heading = ((azimuthDeg + 360.0) % 360.0)
   180	
   181	        eventSink?.success(listOf(heading, accuracy.toDouble()))
   182	    }
