   172	  double _calculateCircularMedian(List<double> samples) {
   173	    if (samples.isEmpty) return 0.0;
   174	    if (samples.length == 1) return samples[0];
   175	
   176	    samples.sort();
   177	
   178	    double maxGap = 0;
   179	    int maxGapIndex = -1;
   180	
   181	    for (int i = 0; i < samples.length - 1; i++) {
   182	      final gap = samples[i + 1] - samples[i];
   183	      if (gap > maxGap) {
   184	        maxGap = gap;
   185	        maxGapIndex = i;
   186	      }
   187	    }
   188	
   189	    final wrapAroundGap = (samples.first + 360) - samples.last;
   190	    if (wrapAroundGap > maxGap) {
   191	      maxGap = wrapAroundGap;
   192	      maxGapIndex = samples.length - 1;
   193	    }
   194	
   195	    List<double> shiftedSamples;
   196	    if (maxGapIndex == samples.length - 1) {
   197	      shiftedSamples = List.from(samples);
   198	    } else {
   199	      shiftedSamples = [];
   200	      final shiftPoint = samples[maxGapIndex];
   201	      for (final s in samples) {
   202	        if (s > shiftPoint) {
   203	          shiftedSamples.add(s);
   204	        } else {
   205	          shiftedSamples.add(s + 360);
   206	        }
   207	      }
   208	    }
   209	
   210	    double median;
   211	    int mid = shiftedSamples.length ~/ 2;
   212	    if (shiftedSamples.length % 2 == 1) {
   213	      median = shiftedSamples[mid];
   214	    } else {
   215	      median = (shiftedSamples[mid - 1] + shiftedSamples[mid]) / 2.0;
   216	    }
   217	
   218	    return median % 360;
   219	  }

   221	  void _updateHeading() {
   222	    final now = DateTime.now().millisecondsSinceEpoch;
   223	    final speedKmh = (state.gpsDataNotifier.value.speed ?? 0) * 3.6;
   224	    final gpsBearing = state.gpsBearingNotifier.value;
   225	
   226	    bool useGps = false;
   227	
   228	    switch (state.compassMode) {
   229	      case CompassMode.magnetic:
   230	        useGps = false;
   231	        break;
   232	      case CompassMode.gps:
   233	        useGps = gpsBearing != null;
   234	        break;
   235	      case CompassMode.auto:
   236	        useGps = gpsBearing != null && speedKmh >= state.autoSwitchSpeedKmh;
   237	        break;
   238	    }
   239	
   240	    state.isGpsCompassActiveNotifier.value = useGps;
   241	
   242	    double newHeading;
   243	
   244	    if (useGps) {
   245	      // GPS-компас
   246	      state.gpsBearingSamples.removeWhere(
   247	        (s) => now - s.$2 > state.averagingPeriod,
   248	      );
   249	      if (state.gpsBearingSamples.isEmpty) return;
   250	      final bearings = state.gpsBearingSamples.map((s) => s.$1).toList();
   251	      final medianTrueBearing = _calculateCircularMedian(bearings);
   252	      // Преобразуем истинный курс в магнитный
   253	      newHeading = (medianTrueBearing - state.magneticDeclination + 360) % 360;
   254	    } else {
   255	      // Магнитный компас
   256	      state.headingSamples.removeWhere(
   257	        (s) => now - s.$2 > state.averagingPeriod,
   258	      );
   259	      if (state.headingSamples.isEmpty) return;
   260	      final headings = state.headingSamples.map((s) => s.$1).toList();
   261	      // Данные с сенсора уже магнитные
   262	      newHeading = _calculateCircularMedian(headings);
   263	    }
   264	
   265	    // Сглаживание
   266	    double diff = newHeading - state.filteredHeading;
   267	    if (diff.abs() > 180) diff += (diff > 0) ? -360 : 360;
   268	
   269	    state.filteredHeading += state.smoothingFactor * diff;
   270	    state.filteredHeading = (state.filteredHeading + 360) % 360;
   271	
   272	    state.headingNotifier.value = state.filteredHeading;
   273	  }

   279	  void _calculateWaypointData() {

   302	  Offset screenToImage(Offset screenPoint) {
   303	    if (state.imageSize == null || state.viewportSize == null) {
   304	      return screenPoint;
   305	    }
   306	
   307	    final vp = state.viewportSize!;
   308	    final t = state.transformState;
   309	    final imageSize = state.imageSize!;
   310	
   311	    final center = Offset(vp.width / 2, vp.height / 2);
   312	    final relative = screenPoint - center - t.translation;
   313	
   314	    final angle = -t.rotationRadians;
   315	    final cos = math.cos(angle);
   316	    final sin = math.sin(angle);
   317	    final derotated = Offset(
   318	      relative.dx * cos - relative.dy * sin,
   319	      relative.dx * sin + relative.dy * cos,
   320	    );
   321	
   322	    final unscaled = derotated / t.scale;
   323	    return unscaled + Offset(imageSize.width / 2, imageSize.height / 2);
   324	  }

   327	    if (state.imageSize == null || state.viewportSize == null) {
   328	      return imagePoint;
   329	    }

   331	    final vp = state.viewportSize!;
   332	    final t = state.transformState;
   333	    final imageSize = state.imageSize!;
   334	
   335	    final center = Offset(vp.width / 2, vp.height / 2);
   336	    final local =
   337	        imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   338	    final scaled = local * t.scale;
   339	
   340	    final angle = t.rotationRadians;
   341	    final cos = math.cos(angle);
   342	    final sin = math.sin(angle);
   343	    final rotated = Offset(
   344	      scaled.dx * cos - scaled.dy * sin,
   345	      scaled.dx * sin + scaled.dy * cos,
   346	    );
   347	
   348	    return center + t.translation + rotated;

   691	  void _centerMapOnUser() {
   692	    final imagePoint = state.currentUserImagePoint;
   693	    if (imagePoint == null || state.viewportSize == null) return;
   694	
   695	    final crosshairScreen = _getCrosshairScreenPoint();
   696	    final vp = state.viewportSize!;
   697	    final vpCenter = Offset(vp.width / 2, vp.height / 2);
   698	    final targetTranslation = crosshairScreen - vpCenter;
   699	
   700	    final t = state.transformState;
   701	    final imageSize = state.imageSize;
   702	    if (imageSize == null) return;
   703	
   704	    final local =
   705	        imagePoint - Offset(imageSize.width / 2, imageSize.height / 2);
   706	    final scaled = local * t.scale;
   707	
   708	    final angle = t.rotationRadians;
   709	    final cos = math.cos(angle);
   710	    final sin = math.sin(angle);
   711	    final rotated = Offset(
   712	      scaled.dx * cos - scaled.dy * sin,
   713	      scaled.dx * sin + scaled.dy * cos,
   714	    );
   715	
   716	    final newTranslation = targetTranslation - rotated;
   717	
   718	    setState(() {
   719	      state.transformState = MapTransformState(
   720	        scale: t.scale,
   721	        rotationRadians: t.rotationRadians,
   722	        translation: newTranslation,
   723	      );
   724	    });
   725	
   726	    _recalculateUserScreenPoint();
   727	  }

   370	  void _onScaleStart(ScaleStartDetails details) {
   371	    _gestureStartTranslation = _state.transformState.translation;
   372	    _gestureStartScale = _state.transformState.scale;
   373	    _gestureStartRotation = _state.transformState.rotationRadians;
   374	    _gestureStartFocalPoint = details.focalPoint;
   375	
   376	    // Запоминаем пиксель под прицелом
   377	    if (_state.viewportSize != null && _state.imageSize != null) {
   378	      final pivotScreen = _logic.getCrosshairScreenPoint();
   379	      _gestureStartPivotImage = _logic.screenToImage(pivotScreen);
   380	    } else {
   381	      _gestureStartPivotImage = null;
   382	    }
   383	
   384	    // Для вращения
   385	    _accumulatedRotation = 0.0;
   386	    if (_state.viewportSize != null && _state.imageSize != null) {
   387	      final pivotScreen = _logic.getCrosshairScreenPoint();
   388	      final startVector = _gestureStartFocalPoint - pivotScreen;
   389	      _lastAngle = math.atan2(startVector.dy, startVector.dx);
   390	    } else {
   391	      _lastAngle = 0.0;

   395	  void _onScaleUpdate(ScaleUpdateDetails details) {
   396	    if (_state.viewportSize == null || _state.imageSize == null) return;
   397	
   398	    final pointerCount = details.pointerCount;
   399	
   400	    // --------------------------------------------------------
   401	    // 1 палец
   402	    // --------------------------------------------------------
   403	    if (pointerCount == 1) {
   404	      if (_state.rotateMode) {
   405	        // Вращение одним пальцем вокруг прицела
   406	        final currentFocalPoint = details.focalPoint;
   407	        final pivotScreen = _logic.getCrosshairScreenPoint();
   408	
   409	        final currentVector = currentFocalPoint - pivotScreen;
   410	        final currentAngle = math.atan2(currentVector.dy, currentVector.dx);
   411	
   412	        double deltaAngle = currentAngle - _lastAngle;
   413	        if (deltaAngle > math.pi) deltaAngle -= 2 * math.pi;
   414	        if (deltaAngle < -math.pi) deltaAngle += 2 * math.pi;
   415	
   416	        _accumulatedRotation += deltaAngle;
   417	        _lastAngle = currentAngle;
   418	
   419	        const sensitivity = 0.8;
   420	        final newRotation = _gestureStartRotation + _accumulatedRotation * sensitivity;
   421	
   422	        if (_gestureStartPivotImage != null) {
   423	          final tempTransform = MapTransformState(
   424	            scale: _state.transformState.scale,
   425	            rotationRadians: newRotation,
   426	            translation: _gestureStartTranslation,
   427	          );
   428	
   429	          final oldTransform = _state.transformState;
   430	          _state.transformState = tempTransform;
   431	
   432	          final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   433	
   434	          _state.transformState = oldTransform;
   435	
   436	          final delta = pivotScreen - pivotScreenAfterRotate;
   437	          final newTranslation = _gestureStartTranslation + delta;
   438	
   439	          _logic.updateTransform(
   440	            MapTransformState(
   441	              scale: _state.transformState.scale,
   442	              rotationRadians: newRotation,
   443	              translation: newTranslation,
   444	            ),
   445	          );
   446	        } else {
   447	          _logic.updateTransform(
   448	            MapTransformState(
   449	              scale: _state.transformState.scale,
   450	              rotationRadians: newRotation,
   451	              translation: _state.transformState.translation,
   452	            ),
   453	          );
   454	        }
   455	      } else {
   456	        // Панорамирование
   457	        final delta = details.focalPoint - _gestureStartFocalPoint;
   458	        final newTranslation = _gestureStartTranslation + delta;
   459	
   460	        _logic.updateTransform(
   461	          MapTransformState(
   462	            scale: _state.transformState.scale,
   463	            rotationRadians: _state.transformState.rotationRadians,
   464	            translation: newTranslation,
   465	          ),
   466	        );
   467	      }
   468	      return;
   469	    }
   470	
   471	    // --------------------------------------------------------
   472	    // 2 пальца — умное масштабирование/вращение
   473	    // --------------------------------------------------------
   474	    if (pointerCount == 2) {
   475	      final scaleChange = (details.scale - 1.0).abs();
   476	      final rotationChange = details.rotation.abs();
   477	
   478	      // Определяем, что преобладает
   479	      if (scaleChange > rotationChange) {
   480	        // ---------- МАСШТАБИРОВАНИЕ ----------
   481	        final newScale = (_gestureStartScale * details.scale).clamp(0.05, 20.0);
   482	
   483	        if (_gestureStartPivotImage != null) {
   484	          final pivotScreen = _logic.getCrosshairScreenPoint();
   485	
   486	          final tempTransform = MapTransformState(
   487	            scale: newScale,
   488	            rotationRadians: _state.transformState.rotationRadians,
   489	            translation: _gestureStartTranslation,
   490	          );
   491	
   492	          final oldTransform = _state.transformState;
   493	          _state.transformState = tempTransform;
   494	
   495	          final pivotScreenAfterScale = _logic.imageToScreen(_gestureStartPivotImage!);
   496	
   497	          _state.transformState = oldTransform;
   498	
   499	          final delta = pivotScreen - pivotScreenAfterScale;
   500	          final newTranslation = _gestureStartTranslation + delta;
   501	
   502	          _logic.updateTransform(
   503	            MapTransformState(
   504	              scale: newScale,
   505	              rotationRadians: _state.transformState.rotationRadians,
   506	              translation: newTranslation,
   507	            ),
   508	          );
   509	        } else {
   510	          _logic.updateTransform(
   511	            MapTransformState(
   512	              scale: newScale,
   513	              rotationRadians: _state.transformState.rotationRadians,
   514	              translation: _gestureStartTranslation,
   515	            ),
   516	          );
   517	        }
   518	      } else {
   519	        // ---------- ВРАЩЕНИЕ ----------
   520	        final pivotScreen = _logic.getCrosshairScreenPoint();
   521	
   522	        // Накопление угла для плавного вращения
   523	        final currentRotation = _gestureStartRotation + details.rotation;
   524	
   525	        if (_gestureStartPivotImage != null) {
   526	          final tempTransform = MapTransformState(
   527	            scale: _state.transformState.scale,
   528	            rotationRadians: currentRotation,
   529	            translation: _gestureStartTranslation,
   530	          );
   531	
   532	          final oldTransform = _state.transformState;
   533	          _state.transformState = tempTransform;
   534	
   535	          final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   536	
   537	          _state.transformState = oldTransform;
   538	
   539	          final delta = pivotScreen - pivotScreenAfterRotate;
   540	          final newTranslation = _gestureStartTranslation + delta;
   541	
   542	          _logic.updateTransform(
   543	            MapTransformState(
   544	              scale: _state.transformState.scale,
   545	              rotationRadians: currentRotation,
   546	              translation: newTranslation,
   547	            ),
   548	          );
   549	        } else {
   550	          _logic.updateTransform(
   551	            MapTransformState(
   552	              scale: _state.transformState.scale,
   553	              rotationRadians: currentRotation,
   554	              translation: _state.transformState.translation,
   555	            ),
   556	          );
   557	        }
   558	      }
   559	      return;
   560	    }
   561	
   562	    // --------------------------------------------------------
   563	    // 3 и более пальцев — вращение вокруг прицела
   564	    // --------------------------------------------------------
   565	    if (pointerCount >= 3) {
   566	      final currentFocalPoint = details.focalPoint;
   567	      final pivotScreen = _logic.getCrosshairScreenPoint();
   568	
   569	      final startVector = _gestureStartFocalPoint - pivotScreen;
   570	      final currentVector = currentFocalPoint - pivotScreen;
   571	
   572	      final startAngle = math.atan2(startVector.dy, startVector.dx);
   573	      final currentAngle = math.atan2(currentVector.dy, currentVector.dx);
   574	
   575	      double deltaAngle = currentAngle - startAngle;
   576	      if (deltaAngle > math.pi) deltaAngle -= 2 * math.pi;
   577	      if (deltaAngle < -math.pi) deltaAngle += 2 * math.pi;
   578	
   579	      const sensitivity = 0.8;
   580	      final newRotation = _gestureStartRotation + deltaAngle * sensitivity;
   581	
   582	      if (_gestureStartPivotImage != null) {
   583	        final tempTransform = MapTransformState(
   584	          scale: _state.transformState.scale,
   585	          rotationRadians: newRotation,
   586	          translation: _gestureStartTranslation,
   587	        );
   588	
   589	        final oldTransform = _state.transformState;
   590	        _state.transformState = tempTransform;
   591	
   592	        final pivotScreenAfterRotate = _logic.imageToScreen(_gestureStartPivotImage!);
   593	
   594	        _state.transformState = oldTransform;
   595	
   596	        final delta = pivotScreen - pivotScreenAfterRotate;
   597	        final newTranslation = _gestureStartTranslation + delta;
   598	
   599	        _logic.updateTransform(
   600	          MapTransformState(
   601	            scale: _state.transformState.scale,
   602	            rotationRadians: newRotation,
   603	            translation: newTranslation,
   604	          ),
   605	        );
   606	      } else {
   607	        _logic.updateTransform(
   608	          MapTransformState(
   609	            scale: _state.transformState.scale,
   610	            rotationRadians: newRotation,
   611	            translation: _state.transformState.translation,
   612	          ),
   613	        );
   614	      }
   615	      return;
   616	    }

    25	  // ---------------------------------------------------------
    26	  // Выбор рабочей пары
    27	  // ---------------------------------------------------------
    28	
    29	  /// Выбирает рабочую пару по правилу:
    30	  /// latest = последняя по времени создания точка
    31	  /// reference = ближайшая предыдущая, расстояние до которой >= 50 м
    32	  MapWorkingPair? selectWorkingPair(
    33	    List<MapAnchor> anchors, {
    34	    double minDistanceMeters = 50.0,
    35	  }) {
    36	    if (anchors.length < 2) return null;
    37	
    38	    final sorted = List<MapAnchor>.from(anchors)
    39	      ..sort((a, b) => a.createdAt.compareTo(b.createdAt));
    40	
    41	    final latest = sorted.last;
    42	
    43	    // Идём от предпоследней точки назад
    44	    for (int i = sorted.length - 2; i >= 0; i--) {
    45	      final candidate = sorted[i];
    46	      final dist = distanceBetweenAnchorsMeters(latest, candidate);
    47	      if (dist >= minDistanceMeters) {
    48	        return MapWorkingPair(latest: latest, reference: candidate);
    49	      }
    50	    }
    51	
    52	    return null;
    53	  }

    59	  double distanceBetweenAnchorsMeters(MapAnchor a, MapAnchor b) {
    60	    return _haversineDistance(a.latitude, a.longitude, b.latitude, b.longitude);
    61	  }

    63	  double _haversineDistance(
    64	    double lat1,
    65	    double lon1,
    66	    double lat2,
    67	    double lon2,
    68	  ) {
    69	    final phi1 = lat1 * math.pi / 180;
    70	    final phi2 = lat2 * math.pi / 180;
    71	    final dPhi = (lat2 - lat1) * math.pi / 180;
    72	    final dLambda = (lon2 - lon1) * math.pi / 180;
    73	
    74	    final a =
    75	        math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    76	        math.cos(phi1) *
    77	            math.cos(phi2) *
    78	            math.sin(dLambda / 2) *
    79	            math.sin(dLambda / 2);
    80	    final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    81	
    82	    return _earthRadius * c;
    83	  }

    93	    required double longitude,
    94	    required double originLat,

   261	  BearingAndDistance bearingAndDistance({
   262	    required double fromLat,
   263	    required double fromLon,
   264	    required double toLat,
   265	    required double toLon,
   266	    required double magneticDeclination,
   267	  }) {
   268	    final dist = _haversineDistance(fromLat, fromLon, toLat, toLon);

   276	        math.cos(phi1) * math.sin(phi2) -
   277	        math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);

   279	    final trueBearing = (math.atan2(y, x) * 180 / math.pi + 360) % 360;
   280	    final magneticBearing = (trueBearing - magneticDeclination + 360) % 360;
   281	
   282	    return BearingAndDistance(
   283	      distanceMeters: dist,
   284	      magneticBearing: magneticBearing,
   285	    );

     1	import 'dart:math' as math;

     3	/// Расчёт координат цели по дистанции и истинному азимуту
     4	Map<String, double> calculateTargetCoordinates({
     5	  required double startLat,
     6	  required double startLon,
     7	  required double distanceMeters,
     8	  required double trueBearingDegrees, // истинный азимут
     9	}) {
    10	  const R = 6371000.0;
    11	
    12	  final phi1 = startLat * math.pi / 180;
    13	  final lambda1 = startLon * math.pi / 180;
    14	  final theta = trueBearingDegrees * math.pi / 180;
    15	  final dR = distanceMeters / R;
    16	
    17	  final sinPhi2 =
    18	      math.sin(phi1) * math.cos(dR) +
    19	      math.cos(phi1) * math.sin(dR) * math.cos(theta);
    20	  final phi2 = math.asin(sinPhi2);
    21	  final y = math.sin(theta) * math.sin(dR) * math.cos(phi1);
    22	  final x = math.cos(dR) - math.sin(phi1) * math.sin(phi2);
    23	  final lambda2 = lambda1 + math.atan2(y, x);
    24	
    25	  return {
    26	    'latitude': phi2 * 180 / math.pi,
    27	    'longitude': lambda2 * 180 / math.pi,
    28	  };
    29	}

    31	double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    32	  const R = 6371e3;
    33	  final phi1 = lat1 * math.pi / 180;
    34	  final phi2 = lat2 * math.pi / 180;
    35	  final dPhi = (lat2 - lat1) * math.pi / 180;
    36	  final dLambda = (lon2 - lon1) * math.pi / 180;
    37	
    38	  final a =
    39	      math.sin(dPhi / 2) * math.sin(dPhi / 2) +
    40	      math.cos(phi1) *
    41	          math.cos(phi2) *
    42	          math.sin(dLambda / 2) *
    43	          math.sin(dLambda / 2);
    44	  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
    45	  return R * c;
    46	}

    49	double calculateTrueBearing(
    50	  double lat1,
    51	  double lon1,
    52	  double lat2,
    53	  double lon2,
    54	) {
    55	  final phi1 = lat1 * math.pi / 180;
    56	  final phi2 = lat2 * math.pi / 180;
    57	  final dLambda = (lon2 - lon1) * math.pi / 180;
    58	
    59	  final y = math.sin(dLambda) * math.cos(phi2);
    60	  final x =
    61	      math.cos(phi1) * math.sin(phi2) -
    62	      math.sin(phi1) * math.cos(phi2) * math.cos(dLambda);
    63	  return (math.atan2(y, x) * 180 / math.pi + 360) % 360;
    64	}

    60	    final colors = [
    61	      Colors.red,
    62	      Colors.green,
    63	      Colors.yellow,
    64	      Colors.cyan,
    65	      Colors.red,

    71	      ..style = PaintingStyle.stroke
    72	      ..strokeWidth = stroke;
    73	
    74	    const sect = math.pi / 4;
    75	    for (int i = 0; i < 8; i++) {
    76	      paint.color = colors[i];
    77	      final start = -math.pi / 2 - sect / 2 + i * sect;
    78	      canvas.drawArc(
    79	        Rect.fromCircle(center: center, radius: radius),
    80	        start,
    81	        sect,
    82	        false,
    83	        paint,
    84	      );
    85	    }

    89	      ..style = PaintingStyle.stroke
    90	      ..strokeWidth = 1.0;
    91	
    92	    const subSect = math.pi / 12; // 15 градусов
    93	    for (int i = 0; i < 8; i++) {
    94	      final sectorColor = colors[i];
    95	      final lineLuminance = sectorColor.computeLuminance();
    96	      linePaint.color = lineLuminance > 0.5
    97	          ? Colors.black.withAlpha(128)
    98	          : Colors.white.withAlpha(128);
    99	
   100	      final majorStart = -math.pi / 2 - sect / 2 + i * sect;
   101	      final line1Angle = majorStart + subSect;
   102	      final line2Angle = majorStart + 2 * subSect;
   103	
   104	      void drawLineAtAngle(double angle) {
   105	        final startPoint = Offset(
   106	          center.dx + (radius - stroke / 2) * math.cos(angle),
   107	          center.dy + (radius - stroke / 2) * math.sin(angle),
   108	        );

   110	          center.dx + (radius + stroke / 2) * math.cos(angle),
   111	          center.dy + (radius + stroke / 2) * math.sin(angle),
   112	        );

   114	      }

   116	      drawLineAtAngle(line1Angle);
   117	      drawLineAtAngle(line2Angle);
   118	    }

   222	  void _drawDashedLine(Canvas canvas, Offset from, Offset to, Paint paint) {
   223	    const dashLength = 10.0;
   224	    const gapLength = 6.0;
   225	    final totalLength = (to - from).distance;
   226	    if (totalLength < 1) return;
   227	
   228	    final direction = (to - from) / totalLength;
   229	    double drawn = 0;
   230	    bool drawing = true;
   231	
   232	    while (drawn < totalLength) {
   233	      final segLen = drawing ? dashLength : gapLength;
   234	      final end = math.min(drawn + segLen, totalLength);
   235	      if (drawing) {
   236	        canvas.drawLine(from + direction * drawn, from + direction * end, paint);
   237	      }
   238	      drawn = end;
   239	      drawing = !drawing;
   240	    }
